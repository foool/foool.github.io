%%Function:BlockInstanceSetup===============================================
%%Abstract:
%%Thisblockallowsonefunctioncallsubsystemasoutput,tobecalled
%%asathreadwakeup.
%%Thisfunction
%%
%%1.CreatesafunctionthatsendsapulseaddsitasabypassfunctiontotheassociatedDAQList
%%2.Createsafunctionthatcontainsthefunctioncallsubsystemcode
%%3.Definesthethreadpriorityforthepulse
%%4.Registersthefunctioncallsubsystemcodewithcodedescriptor
 
%function BlockInstanceSetup(block, system) void
  %addtorecord block AsyncCallerGenCode TLC_TRUE
 
  %% Need to find the downstream function-call subsystems. There is
  %% only one output port on the s-function.
  %% Simulink will have already verified that, if connected at all,
  %% it is a function-call subsystem and that it is the only destination.
  %% The remaining test is whether the destination has other function-call
  %% sources besides this Interrupt block which is not allowed since the
  %% block installs it as an ISR.
  %% That test can done by checking LibGetRecordWidth(ssBlock.ControlInputPort).
  %if "%" == "unconnected"
      %assign wrnTxt = "No code will be generated for the XCP UDP Bypass " ...
           "block since it is not connected to a system."
      %warning %<wrnTxt>
  %endif
 
  %openfile blockdefs
 
    %assign blockname = FEVAL("slrealtime.internal.async.mslrtblockname", "%<Name>")
 
    struct {
        uint32_T coid;
        uint32_T pri;
    } %<blockname>_D;
 
  %closefile blockdefs
  %<LibCacheDefine(blockdefs)>r
 
%endfunction
 
%%Function:Outputs
=============================================================================
%%Abstract:
%%CreatetheISRfunction,mustbeinOutputstogeneratethe
%%correctcode.
%%
 
%function Outputs( block, system ) Output
 
  %assign blockname = FEVAL("slrealtime.internal.async.mslrtblockname", "%<Name>")
 
  %if "%" != "unconnected"
 
    %if system.CurrentTID == -1
        %% Create an ISR to execute the function call
        %openfile funcbuf
        /* SimulinkRealTime Bypass Interrupt Block '%<Name>' */
        // This function is the 'task' that runs inside a thread created
        // in our slrealtime startup code.
 
        void %<blockname>_fc(void)
        {
            %<LibBlockExecuteFcnCall( block, 0 )>
        }
        %closefile funcbuf
        %assign srcFile = LibGetModelDotCFile()
        %<LibSetSourceFileSection(srcFile, "Functions", funcbuf)>
    %endif
 
  %endif
%endfunction
 
%%Function:Start
=============================================================================
%%Abstract:
%%Addthefunction-callsubystemtocodedescriptorandregisteritwiththeassociatedDAQList
 
%function Start( block, system ) Output
    %assign blockname = FEVAL("slrealtime.internal.async.mslrtblockname", "%<Name>")
    %if "%" != "unconnected"
        // Sending the pulse runs the thread at this priority.
        %assign priority = %<SFcnParamSettings.Priority>
        %<blockname>_D.pri = %<priority>;
 
        %assign SSBlock = LibGetFcnCallBlock( block, 0 )
        %addtorecord block ssBlock SSBlock
        %assign tid = block.ssBlock.AsyncShellTID
        %assign TaskID = SampleTime[tid].TaskIdx
        %assign function = "%<blockname>_fc"
        %<FcnAddAsyncFunctionToCodeDescriptor( function, TaskID, priority )>
        // TaskID is %<TaskID>, tid is %<tid> for %<blockname>_fc
         
        %%Note: cannot put the following code in the Start trailer because the code for a function-call
        %% subsystem will still be placed after the trailer. Using the StartCallbackService instead.
        {
            /* XCP UDP Data Acquisition Block (%<ParamSettings.FunctionName>) - '%<Name>' - Start Callback*/
            /* The following code must execute after all daq lists have been configured */
            auto xcpBypassStartCB = []() {
                auto bypassTriggerFcn = std::bind(slrealtime::triggerAsyncSubRate, %<TaskID> );
                xcp::master::bypass::addDAQBypassFcn("%<SFcnParamSettings.DLTag>", bypassTriggerFcn, %<priority>);
            };
            /*CB Priority must be higher than the start cb for the XCP UDP Configuration block*/
            slrealtime::StartCallbackService::registerCB(xcpBypassStartCB, 1);
        }
    %endif
%endfunction
