%function BlockInstanceSetup(block, system) void
    %assign configTag = SFcnParamSettings.ConfigTag
    %assign nodeTag = SFcnParamSettings.NodeTag
    %assign pgn = SFcnParamSettings.Pgn
    %assign dataLength = SFcnParamSettings.DataLength
    %assign nodeAddress = SFcnParamSettings.NodeAddress
    %assign sourceAddr = SFcnParamSettings.SourceAddr
    %assign outputStatus = SFcnParamSettings.OutputStatus
    %assign destinationType = SFcnParamSettings.DestinationType
    %assign startBitPosition = SFcnParamSettings.STARTBITS
    %assign bitLength = SFcnParamSettings.SIGNALSIZES
    %assign gains = SFcnParamSettings.FACTORS
    %assign offsets = SFcnParamSettings.OFFSETS
    %assign maximums = SFcnParamSettings.MAXIMUMS
    %assign minimums = SFcnParamSettings.MINIMUMS
    %assign numSignals = SFcnParamSettings.NumSignals
 
    %assign block = block + configTag + nodeTag + pgn + dataLength ...
        + nodeAddress + sourceAddr + outputStatus + destinationType ...
        + startBitPosition + bitLength + gains + offsets + maximums ...
        + minimums + numSignals
%endfunction
 
%function Start(block, system) Output
{
 
    HCONFIG configHandle;
    #if defined(XPCMSVISUALC) && !defined(MATLAB_MEX_FILE)
    if (!xpcIsModelInit())
    #endif
    {
         configHandle = J1939GetHandle("%<configTag>"); // config_tag: "modelname_configname"
         %<LibBlockPWork(CONFIGHANDLE, "", "", 0)> = (void *)configHandle;
         if(%<LibBlockPWork(CONFIGHANDLE, "", "", 0)> !=NULL)
         {
              j1939Error error;
              PgnEntry_T* pgn_entry_ptr;
              pgn_entry_ptr = J1939RegisterPG( configHandle,
                                                             "%<nodeTag>",
                                                             0,
                                                             %<pgn>,
                                                             RX_PGN,
                                                             %<dataLength>,
                                                             1, /*unused*/
                                                             %<nodeAddress>,
                                                             %<sourceAddr>,
                                                             &error);
 
            %<LibBlockPWork(PGPTR, "", "", 0)> = (void *)pgn_entry_ptr;
            if(pgn_entry_ptr==NULL)
            {
                char msg[256];
                sprintf(msg, "j1939Receive: PGN Registration Error: %d/n",error);
                %%ssSetErrorStatus(S, msg);
                %<RTMSetErrStat("msg")>;
                %<RTMSetStopRequested(1)>;
                return;
            }
         }
     }
}
%endfunction
 
%function Outputs(block, system) Output
{
    /*HCONFIG configHandle = (HCONFIG) %<LibBlockPWork(CONFIGHANDLE, "", "", 0)>; */
    PgnEntry_T* pgn_entry_ptr = (PgnEntry_T *)%<LibBlockPWork(PGPTR, "", "", 0)>;
    uint8_T output_port_offset = 0;
    uint8_T newmsg = 0;
    %assign output_port_offset = 0
    %if (outputStatus)
        %assign output_port_offset = 1
    %endif
     
    if(pgn_entry_ptr!=NULL && pgn_entry_ptr->CA->Active && LOCK_PGN_ENTRY(pgn_entry_ptr,J1939_APP_TASK))
    {
         
        /*Tune Source Address Filter and Destination Type*/
        pgn_entry_ptr->TargetAddress = %<sourceAddr>;
        pgn_entry_ptr->DestinationType = %<destinationType>;
                 
        //printf("RX: Checking PGN(%d)/n",pgn_entry_ptr->PGN.value);
        if(pgn_entry_ptr->NewMessage)
        {
            newmsg = 1;
 
 
            %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
            %assign outputIdx = 0
            %if (SFcnParamSettings.DataLength != 0)
            {
                %assign modeSignalIdx = -1
                %% Check for multiplexor signal by looping over.
                %foreach signalIdx = %<SFcnParamSettings.NumSignals>
                    %assign multiplexType = %
                    %if isMultiplexorSignal(multiplexType)
                        %assign modeSignalIdx = signalIdx
                        %assign modeSignalValue = LibBlockOutputSignal(modeSignalIdx,"","",0)
                        %break
                    %endif
                %endforeach
                %% Run the signals twice to find the mode signal first.
                %foreach pass=2
                    %% Signal unpacking comes here.
                    %foreach signalIdx = %<SFcnParamSettings.NumSignals>
                        %assign startBit = %
                        %assign signalSize = %
                        %assign dataType = %
                        %assign byteOrder = %
                        %assign multiplexType = %
                        %assign multiplexValue = %
                        %assign factor = %
                        %assign offset = %
                        %assign minimum = %
                        %assign maximum = %
 
                        %<checkSignalLength(block, signalSize, dataType, signalIdx, "Unpack")>
 
                        %if pass==0 && isMultiplexedSignal(multiplexType)
                            %% Skip multiplexed signals on first run.
                            %% Do Standard and Multiplexor signals.
                            %continue
                        %elseif pass==1 && !isMultiplexedSignal(multiplexType)
                            %% Skip multiplexor and standard signals
                            %% Do multiplexed signals now.
                            %continue
                        %endif
 
                        %% Generate comments.
                        /* --------------- START Unpacking signal %<signalIdx> ------------------
                        * startBit = %<startBit>
                        * length = %<signalSize>
                        * desiredSignalByteLayout = %<getEndianString(byteOrder)>
                        * dataType = %<getDataTypeString(dataType)>
                        * factor = %<factor>
                        * offset = %<offset>
                        * -----------------------------------------------------------------------*/
                        %% Make sure signal is connected to some other block or multiplexor signal
                        %% as multiplexor signal must always be unpacked.
                        %if LibBlockOutputSignalConnected(signalIdx) || isMultiplexorSignal(multiplexType)
                            %if isMultiplexedSignal(multiplexType) && (modeSignalIdx!=-1)
                                /* Perform the mode dependency check to value */
                                if (%<modeSignalValue> == %<multiplexValue>) {
                            %endif
                            %% Call the unpacking routine.
                            %<setOutputSignal(block, system, "pgn_entry_ptr->DataBufferPtr", signalIdx)>
 
                            %% End of dynamic inclusion of if statement.
                            %if isMultiplexedSignal(multiplexType) && (modeSignalIdx!=-1)
                                }
                            %endif
                        %else
                            /*
                             * Signal is not connected or connected to terminator.
                             * No unpacking code generated.
                            */
                        %endif
                        %% Increment the outputIdx for further port assignments.
                        %assign outputIdx = outputIdx + 1
                    %endforeach %% [signalIdx]
                %endforeach %% [pass]
            }
            %endif
            %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 
            pgn_entry_ptr->NewMessage = 0;
        
        }
        UNLOCK_PGN_ENTRY(pgn_entry_ptr,J1939_APP_TASK);
 
    }
 
    %if (output_port_offset)
    {
        real_T* y = (real_T *) %<LibBlockOutputSignalAddr(numSignals,"","",0)>;
        y[0] = newmsg;
    }
    %endif
}
%endfunction
 
%%Function:Terminate==========================================================
%%
%function Terminate(block, system) Output
{
    #if defined(XPCMSVISUALC) && !defined(MATLAB_MEX_FILE)
    if (!xpcIsModelInit())
    #endif
    {
        /* J1939 Tx Block (%<ParamSettings.FunctionName>) - '%<Name>' - Terminate */
     
        /* Release the PG and config handle. */
        if (%<LibBlockPWork(CONFIGHANDLE, "", "", 0)> != NULL){
           J1939DestroyPG((PgnEntry_T *)%<LibBlockPWork(PGPTR, "", "", 0)>);
           J1939ReleaseHandle((HCONFIG) %<LibBlockPWork(CONFIGHANDLE, "", "", 0)> );
        }
    }
}
%endfunction %% Terminate
