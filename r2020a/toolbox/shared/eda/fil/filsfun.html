%function BlockTypeSetup(block, system) void
    
    %<LibAddToCommonIncludes("filcommon_capi.hpp")>
    %<LibAddToCommonIncludes("stdlib.h")>
    %<LibAddToCommonIncludes("stdio.h")>
     
%endfunction
 
 
%%Function:Start===========================================================
%%
%function Start(block, system) Output
    {
        %assign numInputPorts = LibBlockNumInputPorts(block)
        %assign numOutputPorts = LibBlockNumOutputPorts(block)
        %assign inputDataSetBitwidth=SFcnParamSettings.inputDataSetBitwidth
        %assign inputDataSetBytewidth=SFcnParamSettings.inputDataSetBytewidth
        %assign outputDataSetBitwidth=SFcnParamSettings.outputDataSetBitwidth
        %assign outputDataSetBytewidth=SFcnParamSettings.outputDataSetBytewidth
        %assign processingMode=SFcnParamSettings.processingMode
        %assign inputFrameSize=SFcnParamSettings.inputFrameSize
        %assign outputFrameSize=SFcnParamSettings.outputFrameSize
        %assign inhrule=SFcnParamSettings.inhrule
        %assign value=SFcnParamSettings.value
        %assign blkBaseFrameTime=SFcnParamSettings.blkBaseFrameTime
        %assign blkBaseScalarTime=SFcnParamSettings.blkBaseScalarTime
         
        %assign mdlSTime_frameSize=SFcnParamSettings.mdlSTime_frameSize
        %assign mdlSTime_scalarAsDbl=SFcnParamSettings.mdlSTime_scalarAsDbl
         
 
        %assign inputSTime_frameSize=SFcnParamSettings.inputSTime_frameSize
        %assign inputSTime_scalarAsDbl=SFcnParamSettings.inputSTime_scalarAsDbl
     
 
        %assign outputSTime_frameSize=SFcnParamSettings.outputSTime_frameSize
        %assign outputSTime_scalarAsDbl=SFcnParamSettings.outputSTime_scalarAsDbl
         
 
        %assign connection=SFcnParamSettings.connection
        %assign libName=LibChangeSlashToDoubleSlash(SFcnParamSettings.libName)
        %assign libParams=SFcnParamSettings.libParams
        %assign protoParams=SFcnParamSettings.protoParams
        %assign resetDut=SFcnParamSettings.resetDut
        %assign resetFpga=SFcnParamSettings.resetFpga
        %assign softwareVersion_majorRev=SFcnParamSettings.softwareVersion_majorRev
        %assign softwareVersion_minorRev=SFcnParamSettings.softwareVersion_minorRev
        %assign testmode=SFcnParamSettings.testmode
         
         /* Variable declarations */
        struct filcommon_PortInfoT * InputPortInfoArray;
        struct filcommon_PortInfoT * OutputPortInfoArray;
        struct filcommon_BlockInfoT * BlockInfo_V;
        struct filcommon_SoftwareVersionT * SoftwareVersion_V;
        struct filcommon_Error_StatusAndMsg * Error_StatusAndMsg;
        static char_T ErrorMsg[FIL_CAPI_MAX_ERROR_STR]="";
         
        char connection[]="%<connection>";
        char libName[]="%<libName>";
        char libParams[]="%<libParams>";
        char protoParams[]="%<protoParams>";
 
 
          
         
        /* Variable initializations */
        %if 0 < %<numInputPorts>
            InputPortInfoArray = (struct filcommon_PortInfoT*) malloc(sizeof(struct filcommon_PortInfoT)*%<numInputPorts>);
        %else
            InputPortInfoArray = NULL;
        %endif
 
        %if 0 < %<numOutputPorts>
            OutputPortInfoArray = (struct filcommon_PortInfoT*) malloc(sizeof(struct filcommon_PortInfoT)*%<numOutputPorts>);
        %else
            OutputPortInfoArray = NULL;
        %endif
 
        BlockInfo_V = (struct filcommon_BlockInfoT *) malloc(sizeof(struct filcommon_BlockInfoT));
        SoftwareVersion_V=(struct filcommon_SoftwareVersionT *) malloc(sizeof(struct filcommon_SoftwareVersionT));
        Error_StatusAndMsg=(struct filcommon_Error_StatusAndMsg *)malloc(sizeof(struct filcommon_Error_StatusAndMsg));
 
        /*Initialize Error Status*/
        Error_StatusAndMsg->Status=0;
 
        /*Initialize Block Specific Parameters*/
        BlockInfo_V->numInputPorts=%<LibBlockNumInputPorts(block)>;
        BlockInfo_V->inputDataSetBitwidth=%<inputDataSetBitwidth>;
        BlockInfo_V->inputDataSetBytewidth=%<inputDataSetBytewidth>;
        BlockInfo_V->numOutputPorts=%<LibBlockNumOutputPorts(block)>;
        BlockInfo_V->outputDataSetBitwidth=%<outputDataSetBitwidth>;
        BlockInfo_V->outputDataSetBytewidth=%<outputDataSetBytewidth>;
        BlockInfo_V->processingMode=%<processingMode>;
        BlockInfo_V->inputFrameSize=%<inputFrameSize>;
        BlockInfo_V->outputFrameSize=%<outputFrameSize>;
        BlockInfo_V->overclocking.inhRule=%<inhrule>;
        BlockInfo_V->overclocking.value=%<value>;
        BlockInfo_V->blkBaseFrameTime=%<blkBaseFrameTime>;
        BlockInfo_V->blkBaseScalarTime=%<blkBaseScalarTime>;
         
        BlockInfo_V->mdlSTime_frameSize=%<mdlSTime_frameSize>;
        BlockInfo_V->mdlSTime_scalarAsDbl=%<mdlSTime_scalarAsDbl>;
                         
        BlockInfo_V->inputSTime_frameSize=%<inputSTime_frameSize>;
        BlockInfo_V->inputSTime_scalarAsDbl=%<inputSTime_scalarAsDbl>;
         
        BlockInfo_V->outputSTime_frameSize=%<outputSTime_frameSize>;
        BlockInfo_V->outputSTime_scalarAsDbl=%<outputSTime_scalarAsDbl>;
         
        /*Initialize software version*/
        SoftwareVersion_V->majorRev=%<softwareVersion_majorRev>;
        SoftwareVersion_V->minorRev=%<softwareVersion_minorRev>;
 
        /*Initialize Port Specific Parameters*/
        %foreach portIdx = numInputPorts+numOutputPorts
         
            %if %<portIdx> < %<numInputPorts>
                InputPortInfoArray[%<portIdx>].name = malloc(sizeof("%"));
                sprintf(InputPortInfoArray[%<portIdx>].name,"%");
 
                InputPortInfoArray[%<portIdx>].signedness = %;
                InputPortInfoArray[%<portIdx>].elemBitwidth = %;
                InputPortInfoArray[%<portIdx>].validPhase = %;
                InputPortInfoArray[%<portIdx>].numElems = %;
                InputPortInfoArray[%<portIdx>].elemSize = %;
                InputPortInfoArray[%<portIdx>].buf = %<LibBlockInputSignalAddr(portIdx, "", "", 0)>;
                InputPortInfoArray[%<portIdx>].portSTime_frameSize = %;
                InputPortInfoArray[%<portIdx>].portSTime_scalarAsDbl = %;
            %else
                 
                OutputPortInfoArray[%].name = malloc(sizeof("%"));
                sprintf(OutputPortInfoArray[%].name,"%");
 
                OutputPortInfoArray[%].signedness = %;
                OutputPortInfoArray[%].elemBitwidth = %;
                OutputPortInfoArray[%].validPhase = %;
                OutputPortInfoArray[%].numElems = %;
                OutputPortInfoArray[%].elemSize = %;
                OutputPortInfoArray[%].buf = %<LibBlockOutputSignalAddr(portIdx-numInputPorts, "", "", 0)>;
                OutputPortInfoArray[%].portSTime_frameSize = %;
                OutputPortInfoArray[%].portSTime_scalarAsDbl = %;
                 
            %endif
 
        %endforeach
 
 
  
        /* Store Structs needed in PWork */
        %<LibBlockPWork(filcommon_PortInfo_Inputs,"","",0)> = InputPortInfoArray;
        %<LibBlockPWork(filcommon_PortInfo_Outputs,"","",0)> = OutputPortInfoArray;
        %<LibBlockPWork(filcommon_BlockInfoT,"","",0)> = BlockInfo_V;
        %<LibBlockPWork(filcommon_SoftwareVersion,"","",0)> = SoftwareVersion_V;
        %<LibBlockPWork(filcommon_Error_StatusAndMsg,"","",0)> = Error_StatusAndMsg;
         
         
        /*Create FILComm object*/
        filcommon_create(%<LibBlockDWorkAddr(DWork[0],"","",0)>,Error_StatusAndMsg);
         
        /*Initialize FILComm Object*/
         
        filcommon_initializeAndStart(%<LibBlockDWork(DWork[0],"","",0)>,
                                %<numInputPorts>,
                                InputPortInfoArray,
                                %<numOutputPorts>,
                                OutputPortInfoArray,
                                BlockInfo_V,
                                connection,
                                libName,
                                libParams,
                                protoParams,
                                %<resetDut>,
                                %<resetFpga>,
                                SoftwareVersion_V,
                                %<testmode>,
                                Error_StatusAndMsg);
        if(Error_StatusAndMsg->Status)
        {
            strcpy(ErrorMsg,Error_StatusAndMsg->ErrorMsg); /*Static variable is needed by ssSetErrorStatus*/
            /*ssSetErrorStatus(rtS,ErrorMsg);*/
            %<LibSetRTModelErrorStatus("ErrorMsg")>;
            return;
        }
 
         
 
    }
%endfunction
 
%%Function:Outputs===========================================================
%%
%function Outputs(block, system) Output
    {
        %assign numInputPorts = LibBlockNumInputPorts(block)
        %assign numOutputPorts = LibBlockNumOutputPorts(block)
        %assign testmode=SFcnParamSettings.testmode
     
        struct filcommon_Error_StatusAndMsg * Error_StatusAndMsg;
        static char_T ErrorMsg[FIL_CAPI_MAX_ERROR_STR]="";
 
         
 
        %if 0 < %<numInputPorts>
            uint8_T IsInputHitArray[%<numInputPorts>];
        %else
            uint8_T * IsInputHitArray = NULL;
        %endif
 
        %if 0 < %<numOutputPorts>
            uint8_T IsOutputHitArray[%<numOutputPorts>];
        %else
            uint8_T * IsOutputHitArray= NULL;
        %endif
         
 
        /*Get the ClassID*/
        uint32_T ClassID=(uint32_T) %<LibBlockDWork(DWork[0],"","",0)>;
 
        %if LibIsSFcnSingleRate(block)
 
            real_T slTime = %<LibGetTaskTimeFromTID(block)>;
 
            %foreach portIdx = numInputPorts
                IsInputHitArray[%<portIdx>]=1;
            %endforeach
 
            %foreach portIdx = numOutputPorts
                IsOutputHitArray[%<portIdx>]=1;
            %endforeach
 
        %else
 
            real_T slTime = %<LibGetTaskTime(0)>;
 
            %foreach portIdx = numInputPorts
                %assign input_port = "InputPortIdx%<portIdx>"
                if (%<LibIsSFcnSampleHit(input_port)>)
                {
                    IsInputHitArray[%<portIdx>]=1;
                }
                else
                {
                    IsInputHitArray[%<portIdx>]=0;
                }
            %endforeach
 
            %foreach portIdx = numOutputPorts
                %assign output_port = "OutputPortIdx%<portIdx>"
                if (%<LibIsSFcnSampleHit(output_port)>)
                {
                    IsOutputHitArray[%<portIdx>]=1;
                }
                else
                {
                    IsOutputHitArray[%<portIdx>]=0;
                }
            %endforeach
 
 
        %endif
        
         
        Error_StatusAndMsg= (struct filcommon_Error_StatusAndMsg *) %<LibBlockPWork(filcommon_Error_StatusAndMsg,"","",0)>;
                 
        Error_StatusAndMsg->Status=0;
 
        /*Output FILComm object*/
        filcommon_outputs(ClassID,%<numInputPorts>,IsInputHitArray,%<numOutputPorts>,IsOutputHitArray,slTime,%<testmode>,Error_StatusAndMsg);
        if(Error_StatusAndMsg->Status)
        {
            strcpy(ErrorMsg,Error_StatusAndMsg->ErrorMsg); /*Static variable is needed by ssSetErrorStatus*/
            /*ssSetErrorStatus(rtS,ErrorMsg);*/
            %<LibSetRTModelErrorStatus("ErrorMsg")>;
            return;
        }
 
 
    }
%endfunction
 
%%Function:Update===========================================================
%%
%function Update(block, system) Output
    {
        struct filcommon_Error_StatusAndMsg * Error_StatusAndMsg;
        static char_T ErrorMsg[FIL_CAPI_MAX_ERROR_STR]="";
        /*Get the ClassID*/
        uint32_T ClassID=(int32_T) %<LibBlockDWork(DWork[0],"","",0)>;
 
         
        Error_StatusAndMsg= (struct filcommon_Error_StatusAndMsg *) %<LibBlockPWork(filcommon_Error_StatusAndMsg,"","",0)>;
         
        Error_StatusAndMsg->Status=0;
 
        filcommon_update(ClassID,Error_StatusAndMsg);
        if(Error_StatusAndMsg->Status)
        {
            strcpy(ErrorMsg,Error_StatusAndMsg->ErrorMsg); /*Static variable is needed by ssSetErrorStatus*/
            /*ssSetErrorStatus(rtS,ErrorMsg);*/
            %<LibSetRTModelErrorStatus("ErrorMsg")>;
            return;
        }
    }
%endfunction
 
%%Function:Terminate===========================================================
%%
%function Terminate(block, system) Output
    {
        %assign numInputPorts = LibBlockNumInputPorts(block)
        %assign numOutputPorts = LibBlockNumOutputPorts(block)
 
        struct filcommon_PortInfoT* InputPortInfoArray;
        struct filcommon_PortInfoT * OutputPortInfoArray;
        struct filcommon_BlockInfoT * BlockInfo_V;
        struct filcommon_SoftwareVersionT * SoftwareVersion_V;
        struct filcommon_Error_StatusAndMsg * Error_StatusAndMsg;
        static char_T ErrorMsg[FIL_CAPI_MAX_ERROR_STR]="";
        /*Get the ClassID*/
        uint32_T ClassID=(uint32_T) %<LibBlockDWork(DWork[0],"","",0)>;
 
 
        InputPortInfoArray = (struct filcommon_PortInfoT*) %<LibBlockPWork(filcommon_PortInfo_Inputs,"","",0)>;
        OutputPortInfoArray = (struct filcommon_PortInfoT*) %<LibBlockPWork(filcommon_PortInfo_Outputs,"","",0)>;
        BlockInfo_V = (struct filcommon_BlockInfoT*) %<LibBlockPWork(filcommon_BlockInfoT,"","",0)>;
        SoftwareVersion_V = (struct filcommon_SoftwareVersionT*) %<LibBlockPWork(filcommon_SoftwareVersion,"","",0)>;
        Error_StatusAndMsg= (struct filcommon_Error_StatusAndMsg *) %<LibBlockPWork(filcommon_Error_StatusAndMsg,"","",0)>;
        Error_StatusAndMsg->Status=0;
 
        if(InputPortInfoArray != NULL){
            /* Free all resources */
            %foreach portIdx = numInputPorts
                free(InputPortInfoArray[%<portIdx>].name);
            %endforeach
            free(InputPortInfoArray);
            InputPortInfoArray=NULL;
        }
 
        if(OutputPortInfoArray != NULL){
            /* Free all resources */
            %foreach portIdx = numOutputPorts
                free(OutputPortInfoArray[%<portIdx>].name);
            %endforeach
            free(OutputPortInfoArray);
            OutputPortInfoArray=NULL;
        }
 
        if(BlockInfo_V != NULL){
            /* Free all resources */
            free(BlockInfo_V);
            BlockInfo_V=NULL;
        }
 
        if(SoftwareVersion_V != NULL){
            /* Free all resources */
            free(SoftwareVersion_V);
            SoftwareVersion_V=NULL;
        }
 
        filcommon_terminate(ClassID,Error_StatusAndMsg);
        if(Error_StatusAndMsg->Status)
        {
            strcpy(ErrorMsg,Error_StatusAndMsg->ErrorMsg); /*Static variable is needed by ssSetErrorStatus*/
            /*ssSetErrorStatus(rtS,ErrorMsg);*/
            %<LibSetRTModelErrorStatus("ErrorMsg")>;
            return;
        }
 
        if(Error_StatusAndMsg != NULL){
            /* Free all resources */
            free(Error_StatusAndMsg);
            Error_StatusAndMsg=NULL;
        }
 
    }
%endfunction
