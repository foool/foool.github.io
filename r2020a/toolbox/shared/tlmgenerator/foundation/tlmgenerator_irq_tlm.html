%%IRQSignalDeclaration
%openfile tlmg_irq_sig_decl
       sc_core::sc_out m_int_op; // interrupt port; active high
%closefile tlmg_irq_sig_decl
 
%%IRQSignalCtor
%openfile tlmg_irq_sig_ctor
  m_int_op("m_int_op"),
%closefile tlmg_irq_sig_ctor
 
%%IRQSignalInitialization
%openfile tlmg_irq_sig_init
  m_int_op.initialize(false);
%closefile tlmg_irq_sig_init
 
 
%%IRQSignalTriggerImplementation
%openfile tlmg_irq_sig_trigger
    if(!(m_comstat_reg & INT_DIS_BIT_MASK))
    {
         m_int_op.write(true); // trigger interrupt port;
    }
%closefile tlmg_irq_sig_trigger
 
%%IRQSignalAckImplementation
%openfile tlmg_irq_sig_ack
       m_int_op.write(false); // ack interrupt port;
%closefile tlmg_irq_sig_ack
