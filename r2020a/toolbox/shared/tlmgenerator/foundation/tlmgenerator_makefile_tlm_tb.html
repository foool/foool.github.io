OBJECTS = obj/mw_support_tb.o obj/%<tbname>.o obj/%<tbname>_main.o

####### Tools
 
AS = as
CC = gcc
CXX = gcc
LD = gcc
AR = ar
RANLIB = ranlib

####### Inc path

TLM_INCPATH = '%<tlmg_config.tlmgTLMIncludePath>'
MATLAB_INCPATH = '%<tlmg_config.tlmgMatlabIncludePath>'
 
####### Lib path
 
SYSTEMC_LIBPATH = '%<tlmg_config.tlmgSystemCLibPath>'
MATLAB_LIBPATH = '%<tlmg_config.tlmgMatlabLibPathGlnx>'
MATLAB_LIBPATH2 = '%<tlmg_config.tlmgMatlabLibPathGlnx2>'
 
SYSTEMC_LIBNAME = '%<tlmg_config.tlmgSystemCLibName>'
 
####### Include, Lib
 
INCDIR = -I ./include -I '../%<tlmname>/include' -I '../%<rtwname>/include' -I '../%<rtwname>/utils' -I $(SYSTEMC_INCPATH) -I $(TLM_INCPATH) -I $(MATLAB_INCPATH)
LIBDIR = -L $(MATLAB_LIBPATH) -Wl,-rpath $(MATLAB_LIBPATH) -Wl,-rpath $(MATLAB_LIBPATH2)

LOG_OPTION = -DTLMG_CURR_VERBOSITY=%<tlmg_config.tlmgTbVerbosity>
 
####### Flags

CFLAGS = $(OPT_CXXFLAGS) $(EXTRA_CXXFLAGS) -DRT -DUSE_RTMODEL -Wall $(INCDIR) $(SC_OPTION) $(LOG_OPTION)
CXXFLAGS = $(OPT_CXXFLAGS) $(EXTRA_CXXFLAGS) -DRT -DUSE_RTMODEL -Wall -Wno-deprecated $(INCDIR) $(SC_OPTION) $(LOG_OPTION)
LDFLAGS = $(LIBDIR) -Wl,--start-group $(LIBS) -Wl,--end-group -lpthread -lm -lstdc++
ARFLAGS = cru
 
####### Implicit rules
 
.SUFFIXES: .cpp .cxx .cc .C .c .s
 
%<"obj/%.o: src/%.s">
    $(AS) $(ASFLAGS) -o $@ $<
 
%<"obj/%.o: src/%.cpp">
    $(CXX) -c $(CXXFLAGS) -o $@ $<
 
%<"obj/%.o: src/%.cxx">
    $(CXX) -c $(CXXFLAGS) -o $@ $<
 
%<"obj/%.o: src/%.cc">
    $(CXX) -c $(CXXFLAGS) -o $@ $<
 
%<"obj/%.o: src/%.C">
    $(CXX) -c $(CXXFLAGS) -o $@ $<
 
%<"obj/%.o: src/%.c">
    $(CC) -c $(CFLAGS) -o $@ $<
 
####### Targets
all:
    $(MAKE) -f makefile.gnu all-am OPT_CXXFLAGS="-O3"
 
all-debug:
    $(MAKE) -f makefile.gnu all-am OPT_CXXFLAGS="-g"
 
relink:
    $(MAKE) -f makefile.gnu relink-am OPT_CXXFLAGS="-O3"
 
relink-debug:
    $(MAKE) -f makefile.gnu relink-am OPT_CXXFLAGS="-g"
 
all-am: dep $(EXEC)
 
dep:
    cd ../%<tlmname>; $(MAKE) -f makefile.gnu all-am
 
$(EXEC): $(OBJECTS) $(STATICLIB)
    $(LD) $(OBJECTS) $(LDFLAGS) -o $(EXEC)
    @echo "-- Build %<tbname>.exe completed --"
    @echo ""
 
relink-am:
    rm -f $(EXEC)
    $(MAKE) -f makefile.gnu all-am
         
clean:
    cd ../%<tlmname>; $(MAKE) -f makefile.gnu clean
    rm -f $(OBJECTS) $(EXEC)
    @echo "-- Clean %<tbname> completed --"
    @echo ""
     
####### Compile dependencies
 
$(OBJECTS): $(HEADERS)

    $(MAKE) /nologo /f makefile.mk relink-am OPT_CXXFLAGS="/O2 /MT /D _NDEBUG" OPT_LDFLAGS=""
 
relink-debug:

    @echo ""

    @echo ""
