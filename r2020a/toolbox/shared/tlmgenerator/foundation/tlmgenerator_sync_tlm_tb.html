%selectfile NULL_FILE
 
%openfile tlmg_sync_tmode_setupcall_tb
    // ------------------------------------------------------
    // Set core TLM component timing mode: Untimed or Timed.
    // ------------------------------------------------------
    // Set whether the target should return delays based on the specified timing
    // parameters in the 'TLM Generation' options or to just return '0' for all
    // transaction and algorithm delays.
    //
    // In your TLM environment, this can be dynamically programmed during
    // simulation execution, but for this testbench we set it just once at the
    // beginning of the simulation based on the 'TLM Testbench' configset
    // options.
    //
    %if ISEQUAL(tlmg_config.tlmgRuntimeTimingMode, "With timing")
    // m_backdoorcfg->SetTimingParam(mw_backdoorcfg_IF::UNTIMED);
       m_backdoorcfg->SetTimingParam(mw_backdoorcfg_IF::TIMED);
    %else
       m_backdoorcfg->SetTimingParam(mw_backdoorcfg_IF::UNTIMED);
    // m_backdoorcfg->SetTimingParam(mw_backdoorcfg_IF::TIMED);
    %endif
%closefile tlmg_sync_tmode_setupcall_tb
     
%openfile tlmg_sync_syncobj_setupcall_tb
    // The timing mode also needs to be set up in a testbench support object.
    //
    // With timing Testbench timing mode
    // ----------- ---------------------
    // No Untimed
    // Yes LooselyTimed
    //
    // Untimed : Sync to '0' delays after every transaction and for
    // every timed wait.
    // Loosely Timed : Sync to the delays returned from the target after
    // every transaction and use non-zero times for timed
    // waits.
    //
    %if ISEQUAL(tlmg_config.tlmgRuntimeTimingMode, "Without timing")
       mw_sync_tb sync(m_utils, mw_sync_tb::Untimed);
    // mw_sync_tb sync(m_utils, mw_sync_tb::LooselyTimed);
    %else
    // mw_sync_tb sync(m_utils, mw_sync_tb::Untimed);
       mw_sync_tb sync(m_utils, mw_sync_tb::LooselyTimed);
    %endif
%closefile tlmg_sync_syncobj_setupcall_tb
 
%openfile tlmg_sync_tdrcall_tb
        // --------------------------------------------------
        // Wait until there is data to read.
        // --------------------------------------------------
        // We can wait for data ready by either polling the INTR bit in the CSR,
        // or by waiting for the IRQ signal attached to the m_int_ip port of the
        // testbench. If neither is available, we can only do a timed wait.
        //
        %if ISEQUAL(tlmg_config.tlmgIrqPortOnOff, "on")
        // Using IRQ:
           sync.syncToDataReady(m_int_ip);
        //
        // Using CSR:
        // m_utils.printMsg(TLMG_PRINT_SYNC_TO_FUNCS,
        // "##/t polling status bit in register for data ready in output buffer.../n");
        // bool sawTimeout = csr.pollForStatus(INT_STAT_BIT_MASK, INT_STAT_BIT_MASK, csreg,
        // 1000, sc_time(1000, SC_NS));
        // if (sawTimeout) {
        // m_utils.printErr(TLMG_ERR_SYNC_TIMEOUT,
        // "## ERROR: Timed-out waiting for buffer non-empty status./n");
        // break;
        // }
        %elseif ISEQUAL(tlmg_config.tlmgCommandStatusRegOnOffInoutput, "on")
        // Using IRQ:
        // sync.syncToDataReady(m_int_ip);
        //
        // Using CSR:
           m_utils.printMsg(TLMG_PRINT_SYNC_TO_FUNCS,
                   "##/t polling status bit in register for data ready in output buffer.../n");
           bool sawTimeout = csr.pollForStatus(INT_STAT_BIT_MASK, INT_STAT_BIT_MASK, csreg,
                                               1000, sc_time(1000, SC_NS));
           if (sawTimeout) {
               m_utils.printErr(TLMG_ERR_SYNC_TIMEOUT,
                   "## ERROR: Timed-out waiting for buffer non-empty status./n");
               break;
           }
        %else
          %if ISEQUAL(tlmg_config.tlmgRuntimeTimingMode, "Without timing")
          // Since we do not have an IRQ or CSR and we are untimed we must allow
          // the target thread to get through its synchronization points such that data
          // is ready. This means to do two waits of zero time.
          //sync.syncToExplicitTime(SC_ZERO_TIME);
          //sync.syncToExplicitTime(SC_ZERO_TIME);
           m_utils.printErr(TLMG_ERR_NOT_SUPPORTED,
               "## ERROR: Untimed simulation requires explicite synchronization like interrupt signal or command and status register./n");
           break;
          %else
          // Since we do not have an IRQ or CSR, we do a timed wait.
            %if ISEQUAL(tlmg_config.tlmgProcessingType,"Periodic SystemC Thread")
            sync.syncToExplicitTime(sc_core::sc_time(%<tlmg_info.PortRate>, sc_core::SC_SEC));
            %else
            sync.syncToExplicitTime(sc_time(ALG_PROC_TIME * 2, SC_NS));
            %endif
          %endif
        %endif
%closefile tlmg_sync_tdrcall_tb
