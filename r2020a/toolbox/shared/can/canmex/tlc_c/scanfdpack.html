%function Start(block, system) Output
     
%endfunction
 
%function Outputs(block, system) Output
    %% Automatic comments generated.
    /* %<SLibBlkType(block)>: '%<SLibBlkName(block)>' */
 
    %assign inAddr = LibBlockInputSignalAddr(0,"","",0)
    %assign canmsg = LibBlockOutputSignal(0,"","",0)
    %assign isRemoteFrame = !SFcnParamSettings.ISNOTREMOTEFRAME
 
    %% Apply the settings for length, ID and Extended field in the Ouputs function
    %<LibBlockOutputSignal(0,"","",0)>.ID = %<SFcnParamSettings.ID>;
    %<LibBlockOutputSignal(0,"","",0)>.Length = %<SFcnParamSettings.LENGTH>;
    %<LibBlockOutputSignal(0,"","",0)>.Extended = %<SFcnParamSettings.EXTENDED>;
    %<LibBlockOutputSignal(0,"","",0)>.Remote = %<isRemoteFrame>;
    %<LibBlockOutputSignal(0,"","",0)>.BRS = %<SFcnParamSettings.ENABLEBRS>;
    %<LibBlockOutputSignal(0,"","",0)>.ProtocolMode = %<SFcnParamSettings.PROTOCOLMODE>;
    %<LibBlockOutputSignal(0,"","",0)>.DLC = %<SFcnParamSettings.DLC>;
     
    %% Initialize data to zero.
    %foreach index = 64
        %<LibBlockOutputSignal(0,"","",0)>.Data[%<index>] = 0;
    %endforeach
     
    %if SFcnParamSettings.ISNOTREMOTEFRAME
    {
        %if SFcnParamSettings.ISRAWDATAFORMAT
                %assign portWidth = LibBlockInputSignalWidth(0)
                %assign dType = LibBlockInputSignalDataTypeName(0,"")
                %assign bytesToCopy = "%<portWidth> * sizeof(%<dType>)"
                (void) %<LibGenMemFcnCall("memcpy", "(%<canmsg>.Data)", inAddr, bytesToCopy)>;
        %else
            %assign modeSignalIdx = -1
 
            %% Check for multiplexor signal by looping over.
            %foreach signalIdx = %<SFcnParamSettings.NUMSIGNALS>
                %assign multiplexType = %
                %if isMultiplexorSignal(multiplexType)
                    %assign modeSignalIdx = signalIdx
                    %break
                %endif
            %endforeach
 
            %if (modeSignalIdx != -1)
                %% @TODO: The type of mode signal value need to be checked.
                int32_T modeSignalValue = 0;
            %endif
 
            %if (modeSignalIdx != -1)
                %% if Mode signal is not connected, do not do anything.
                %if LibBlockInputSignalConnected(modeSignalIdx)
                    %% Pack mode signal.
                    %assign startBit = %
                    %assign signalSize = %
                    %assign dataType = %
                    %assign byteOrder = %
                    %assign factor = %
                    %assign offset = %
                    %assign minimum = %
                    %assign maximum = %
 
                    %<checkSignalLength(block, signalSize, dataType, modeSignalIdx, "Pack")>
 
                    /* --------------- START Packing mode signal %<modeSignalIdx> ------------------
                    * startBit = %<startBit>
                    * length = %<signalSize>
                    * desiredSignalByteLayout = %<getEndianString(byteOrder)>
                    * dataType = %<getDataTypeString(dataType)>
                    * factor = %<factor>
                    * offset = %<offset>
                    * minimum = %<minimum>
                    * maximum = %<maximum>
                    * -----------------------------------------------------------------------*/
                    %<setOutputMessage(block, system, "%<canmsg>.Data", modeSignalIdx, "CAN")>
                     
                    %% Unpack the mode signal from the message.
                    %assign rawDataType = "%<getDataType(dataType, signalSize)>"
                    %<unpackSignal(block, system, "%<canmsg>.Data", modeSignalIdx, "int32_T", startBit, /
                                                    signalSize, byteOrder, "modeSignalValue", rawDataType)>
 
                %else
                    %% The mode signal is either not connected or connected to
                    %% ground.
                %endif
            %endif
 
            %% Call Signal Packing
            %foreach signalIdx = %<SFcnParamSettings.NUMSIGNALS>
                %assign startBit = %
                %assign signalSize = %
                %assign dataType = %
                %assign byteOrder = %
                %assign multiplexType = %
                %assign multiplexValue = %
                %assign factor = %
                %assign offset = %
                %assign minimum = %
                %assign maximum = %
                 
                %<checkSignalLength(block, signalSize, dataType, signalIdx, "Pack")>
                 
                %if isMultiplexorSignal(multiplexType) && (modeSignalIdx!=-1)
                    %%Since mode signal has been packed, just continue.
                    %continue
                %endif
 
                /* --------------- START Packing signal %<signalIdx> ------------------
                * startBit = %<startBit>
                * length = %<signalSize>
                * desiredSignalByteLayout = %<getEndianString(byteOrder)>
                * dataType = %<getDataTypeString(dataType)>
                * factor = %<factor>
                * offset = %<offset>
                * minimum = %<minimum>
                * maximum = %<maximum>
                * -----------------------------------------------------------------------*/
 
                %% Call function set the output CAN message.
                %if LibBlockInputSignalConnected(signalIdx)
                    %if isMultiplexedSignal(multiplexType) && (modeSignalIdx!=-1)
                        if (modeSignalValue == %<multiplexValue>){
                    %endif
                    %<setOutputMessage(block, system, "%<canmsg>.Data", signalIdx, "CAN")>
                    %if isMultiplexedSignal(multiplexType) && (modeSignalIdx!=-1)
                        }
                    %endif
                %endif
            %endforeach %% [signalIdx]
        %endif
    }
    %endif
%endfunction
