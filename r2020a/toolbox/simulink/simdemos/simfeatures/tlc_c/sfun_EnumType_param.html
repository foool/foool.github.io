%%Function:mdlOutputs=======================================================
%%Abstract:
%%y=u*Gain+Offset
%%
%function Outputs(block, system) Output
  /* %<Type> Block: %<Name> */
 {
  %assign offset = LibBlockParameter(Offset, "", "", 0)
  %assign gain = SFcnParamSettings.gain
  %assign rollVars = ["U", "Y"]
  %roll idx = RollRegions, lcv = RollThreshold, block, "Roller", rollVars
    %<LibBlockOutputSignal(0, "", lcv, idx)> = /
    %<LibBlockInputSignal(0, "", lcv, idx)> * %<gain> + %<offset>;
  %endroll
 }
 
%endfunction
