%implements "sfun_varsize_signalWidth" "C"
%%Function:Outputs===========================================================
%%Abstract:
%%setoneoutputastheinputsignalandtheotherasthedimensionoftheinputvalue
%%
%function Outputs(block, system) Output
/* %<Type> Block: %<Name> */
%assign iPort = LibBlockParameterValue(block.Parameter[0], 0)
%assign outSig1 = LibBlockOutputSignal(0, "", "", 0)
%assign outSig2 = LibBlockOutputSignal(1, "", "", 0)
%assign numDims = LibBlockInputSignalNumDimensions(0)
{
  int iPortWidths = 1;
  int i;
  %assign inSig = LibBlockInputSignal(%<iPort>, "", "", 0)
 
  %foreach dimIdx = numDims
    %assign cdm = SLibGetCurrentInputPortDimensions(%<iPort>, dimIdx)
    iPortWidths = %<cdm> * iPortWidths;
  %endforeach
  %<outSig2> = iPortWidths;
 
  for(i=0; i    (&%<outSig1>)[i] = (&%<inSig>)[i];
  }
}
%endfunction %% Outputs
