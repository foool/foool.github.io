%implements "sfun_pwm" "C"
 
%function Start(block, system) Output
  %assign d1 = LibBlockDWork(DWork[0], "", "", 0)
  %assign d2 = LibBlockDWork(DWork[1], "", "", 0)
  %assign d3 = LibBlockDWork(DWork[2], "", "", 0)
  %<d1> = true;
  %<d2> = 0.0;
  %<d3> = 0;
  %<RTMSetNumTicksToNextHitForCtrlRate(block, 0, "1")>;
%endfunction
 
%function Outputs(block, system) Output
  %assign y = LibBlockOutputSignal(0, "", "", 0)
  %assign d1 = LibBlockDWork(DWork[0], "", "", 0)
  %assign d2 = LibBlockDWork(DWork[1], "", "", 0)
  %assign d3 = LibBlockDWork(DWork[2], "", "", 0)
   
  size_t numTicksToNextHit = 0;
  real_T yout_value = 0.0;
   
  if (%<d1>) {
    %assign amplitude = LibBlockParameterValue(block.Parameter[0], 0)
    %assign resolution = LibBlockParameterValue(block.Parameter[1], 0)
    %assign duty = LibBlockInputSignal(0, "", "", 0)
    %assign period = LibBlockInputSignal(1, "", "", 0)
    size_t numSamples = (size_t) (%<period> / %<resolution>);
    numTicksToNextHit = (size_t) (%<duty> * numSamples);
    if (numTicksToNextHit == 0) {
      numTicksToNextHit = numSamples;
      yout_value = 0.0;
    } else if (numTicksToNextHit == numSamples) {
      numTicksToNextHit = numSamples;
      yout_value = %<amplitude>;
    } else {
      %<d1> = false;
      %<d2> = %<duty>;
      %<d3> = numSamples - numTicksToNextHit;
      yout_value = %<amplitude>;
    }
  } else {
    %<d1> = true;
    numTicksToNextHit = (size_t) %<d3>;
    yout_value = 0.0;
  }
   
  %<y> = yout_value;
  
  %<RTMSetNumTicksToNextHitForCtrlRate(block, 0, "numTicksToNextHit")>;
   
%endfunction
