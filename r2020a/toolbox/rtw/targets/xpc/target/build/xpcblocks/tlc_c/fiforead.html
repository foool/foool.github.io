%implements fiforead "C"
 
 
%function Start(block, system) Output
  %assign frSettings = block.SFcnParamSettings.settings
  %<LibBlockIWork("", "", "", 0)> = %<frSettings[0]>; /* Max to read */
  %<LibBlockIWork("", "", "", 1)> = %<frSettings[1]>; /* Min to read */
  %<LibBlockIWork("", "", "", 2)> = %<frSettings[2]>; /* Use delimiter? */
  %<LibBlockIWork("", "", "", 3)> = %<frSettings[3]>; /* delimiter */
  %<LibBlockIWork("", "", "", 4)> = %<frSettings[4]>; /* FIFO type */
 
  /* Input pointer(s) */
  %<LibBlockPWork("ioArgs", "", "", 0)> = %<LibBlockInputSignalAddr(0, "", "", 0)>;
  %if ISEQUAL(%<LibBlockNumInputPorts(block)>, 1)
    %<LibBlockPWork("", "", "", 1)> = NULL;
    %<LibBlockPWork("", "", "", 2)> = NULL;
  %else
    %<LibBlockPWork("", "", "", 1)> = %<LibBlockInputSignalAddr(1, "", "", 0)>;
    %<LibBlockPWork("", "", "", 2)> = %<LibBlockInputSignalAddr(2, "", "", 0)>;
  %endif
 
  /* Output pointer(s) */
  %<LibBlockPWork("", "", "", 3)> = %<LibBlockOutputSignalAddr(0, "", "", 0)>;
  %if ISEQUAL(%<LibBlockNumOutputPorts(block)>, 1)
    %<LibBlockPWork("", "", "", 4)> = NULL;
  %else
    %<LibBlockPWork("", "", "", 4)> = %<LibBlockOutputSignalAddr(1, "", "", 0)>;
  %endif
%endfunction
 
%% Function: BlockTypeSetup ===============================================
%%
%function BlockTypeSetup(block, system) void
    %<LibAddToModelSources("fiforeadwrite")>
    %openfile buf
    extern void fiforead(const void *iPorts[], void *oPorts[], int *settings);
    %closefile buf
    %<LibCacheFunctionPrototype(buf)>
%endfunction
 
%function Outputs(block, system) Output
{
  void **iPtrs = &%<LibBlockPWork("ioArgs", "", "", 0)>;
  void **oPtrs = &%<LibBlockPWork("ioArgs", "", "", 3)>;
  fiforead(iPtrs, oPtrs, &%<LibBlockIWork("settings", "", "", 0)>);
}
%endfunction
