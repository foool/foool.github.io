%implements xpcbytepacking "C"
 
%function Outputs(block, system) Output
    %assign direction = block.SFcnParamSettings.Direction
    %assign bytes = block.SFcnParamSettings.Bytes
    %assign index = block.SFcnParamSettings.Index
 
    %switch direction
 
        %case 0 %% PACK
            /* Byte Packing: %<block.Name> */
            %foreach j = NumDataInputPorts
                (void)memcpy((uint8_T*)%<LibBlockOutputSignalAddr(0, "", "", 0)> + %<index[j]>, ...
                             (uint8_T*)%<LibBlockInputSignalAddr(j, "", "", 0)>, ...
                             %<bytes[j]>);
            %endforeach
        %break
 
        %case 1 %% UNPACK
            /* Byte Unpacking: %<block.Name> */
            %foreach j = NumDataOutputPorts
                (void)memcpy((uint8_T*)%<LibBlockOutputSignalAddr(j, "", "", 0)>, ...
                             (uint8_T*)%<LibBlockInputSignalAddr(0, "", "", 0)> + %<index[j]>, ...
                             %<bytes[j]>);
            %endforeach
            %break
 
    %endswitch %% direction
 
%endfunction %% xpcbytepacking
