%/
 
 
%function xPCTypesDoOnce() void
  %if EXISTS("::xPCTypesSetupDone") && ::xPCTypesSetupDone == 1
    %return
  %endif
  %assign ::xPCTypesSetupDone = 1
 
  %with CompiledModel.DataTypes
      %foreach dtypeIdx = NumDataTypes
          %assign dtype = DataType[dtypeIdx]
          %if (ISEQUAL(dtype.Name, "serialfifoptr") )
              %% The variable name is serialfifoground and the value
              %% to use is the external serialfifoground.
              %<SLibSetDataTypeGroundName(dtype, "serialfifoground")>
              %<SLibSetDataTypeGroundValue(dtype, "serialfifoground")>
              %% Use the definition in xpc_datatype_ground.c, below.
              %addtorecord dtype GroundAlreadyDefined TLC_TRUE
          %endif
          %% similar for the 3 1553 datatypes
          %if (ISEQUAL(dtype.Name, "bcmsglist1553") )
              %<SLibSetDataTypeGroundName(dtype, "bcmsg1553ground")>
              %<SLibSetDataTypeGroundValue(dtype, "bcmsg1553ground")>
              %addtorecord dtype GroundAlreadyDefined TLC_TRUE
          %endif
          %if (ISEQUAL(dtype.Name, "bcstatus1553") )
              %<SLibSetDataTypeGroundName(dtype, "bcstatground")>
              %<SLibSetDataTypeGroundValue(dtype, "bcstatground")>
              %addtorecord dtype GroundAlreadyDefined TLC_TRUE
          %endif
          %if (ISEQUAL(dtype.Name, "bmmsglist1553") )
              %<SLibSetDataTypeGroundName(dtype, "bmmsg1553ground")>
              %<SLibSetDataTypeGroundValue(dtype, "bmmsg1553ground")>
              %addtorecord dtype GroundAlreadyDefined TLC_TRUE
          %endif
      %endforeach
  %endwith
  %% Make ground reference for all the custom data types
  %openfile buffer = "xpc_datatype_ground.c"
      #include "xpcdatatypes.h"
      const serialfifoptr serialfifoground = {0,0,0};
      %% similar for the 1553 data types
      const bcmsglist1553 bcmsg1553ground = {0,0,0,0};
      const bcstatus1553 bcstatground = {0,0,0,0,0,0};
      const bmmsglist1553 bmmsg1553ground = {0,0,0,0};
  %closefile buffer
  %<LibAddToModelSources("xpc_datatype_ground")>
  %<LibCacheExtern("extern const serialfifoptr serialfifoground;")>
  %<LibCacheExtern("extern const bcmsglist1553 bcmsg1553ground;")>
  %<LibCacheExtern("extern const bcstatus1553 bcstatground;")>
  %<LibCacheExtern("extern const bmmsglist1553 bmmsg1553ground;")>
 
  %<LibAddToCommonIncludes("
  %<LibAddToCommonIncludes("
%endfunction /% xPCTypesDoOnce() %/
 
 
%<xPCTypesDoOnce()>
