/% File: slrtAddArtifacts.tlc
 % Copyright 2017 The MathWorks, Inc.
 % This function calls AddFileToApp.m with a 'first time' flag and with
 % the system, partName and fileName.
 %
 % partName is the path and name on the target machine, relative to the
 % top level application directory, usually C:/private
 %
 % fileName is the location on the host machine to get the file from.
 %/
  
 %function slrtAddArtifacts( system, partName, fileName ) void
   %assign first = 1
   %if (EXISTS("::slrtArtifactsFirst") && ::slrtArtifactsFirst == 1)
      %assign first = 0
   %else
      %assign ::slrtArtifactsFirst = 1
   %endif
 
   %assign tf = FEVAL("slrtPrivate", "AddFileToApp", first, system, partName, fileName )
   %if tf == 0
      %error("Model build error: failure to add %<fileName> to mldatx")
   %endif
 %endfunction
