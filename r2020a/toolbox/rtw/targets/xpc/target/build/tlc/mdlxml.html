%%------------------------------------------------
 
%function genxmlsubsys(subsys) Output
 %foreach sys=subsys.NumChildSubsystems
  %assign subsysIdx = subsys.ChildSubsystems[sys]
  %assign slName = bhm.Subsystem[subsysIdx].SLName
  %assign slName = FEVAL ("repxpcxmlSchars",slName)
  %%Dummy Subsys Node
  %assign xpcName= FEVAL ("repxpcxmlSchars", SLibMangledGrBlockPath(bhm.Subsystem[subsysIdx].SubsystemBlockIndex))
  
  %<slName>
  %<xpcName>
  %<genxmlsubsys(bhm.Subsystem[subsysIdx])>
  %<genxmlsubchilds(Subsystem[subsysIdx],subsysIdx)>
  
 %endforeach %%sys=subsys.NumChildSubsystems
%endfunction

%function genxmlsubchilds(subsys,sys) Output
 
%assign slName = FEVAL ("repxpcxmlSchars", SLibMangledGrBlockPath(subsys.SubsystemBlockIndex))
%if sys == 0
 %assign slName = subsys.SLName
%endif
%assign mdlparam = CompiledModel.ModelParameters
%with subsys
  %if InlineParameters != 1
    %%-params--------------------------
    %foreach blkIdx = NumBlocks
      %with Block[blkIdx]
 %%if Type == "Stateflow"
  %%continue
 %%endif
        %assign niPorts = NumDataInputPorts + NumControlInputPorts
        %assign noPorts = NumDataOutputPorts + NumFcnCallOutputPorts
        %assign isSrcBlk = (noPorts > 0) && (niPorts == 0)
        %% Block Path
        %assign grBlkIdx = [%<sys>, %<blkIdx>]
        %assign count = 0
        %foreach paramIdx = NumParameters
          %assign mParam = Parameter[paramIdx]
          %if mParam._idx < 0 %% post compile virtual blocks
            %continue
          %endif
          %assign param = CompiledModel.ModelParameters.Parameter[mParam._idx]
          %if (param.MemoryMapIdx[2] == -1) || (param.Tunable == "no")
            %% Inaccessible data or non tunable parameter
            %continue
          %elseif (!ISEMPTY(param.WorkspaceVarName))
            %% Parameter represents a workspace variable
            %continue
          %endif
          %%-----Index reference into block Idx and paramidx in system record
          %assign sysIdx = param.ReferencedBy[0][0]
          %assign blkRef = param.ReferencedBy[0][2]
          %assign parRef = param.ReferencedBy[0][3]
          %%-----
          %assign structIdx = param.MemoryMapIdx[0]
          %assign secIdx = param.MemoryMapIdx[1]
          %assign dataIdx = param.MemoryMapIdx[2]
          %if (secIdx == -1)
            %assign data = CompiledModel.GlobalMemoryMap.UnstructuredData.Data[dataIdx]
            %assign isComplex = data.IsComplex
            %assign dataTypeIdx = data.DataTypeIdx
          %else
            %assert(structIdx != -1)
            %assign section=CompiledModel.GlobalMemoryMap.StructuredData[structIdx].Section[secIdx]
            %assign data = section.Data[dataIdx]
            %assign dataTypeIdx= section.DataTypeIdx
            %assign isComplex = section.IsComplex
          %endif
          %if isComplex
            %continue
          %endif
          %assign isReadOnly = data.Permission == "ro"
          %%assign isComputed = CompiledModel.System[sysIdx].Block[blkRef].Parameter[parRef].StringType == "Computed"
          %assign isOtherClass = data.Class == "other"
          %assign isFixedPoint = data.IsFixedPoint
          %%%%%%%%%%
          %% Skip certain kinds of data - Read only
          %if isReadOnly || isOtherClass %%|| (isComputed && !isFixedPoint)
            %continue
          %endif
          %% Address
          %assign addrMapIdx = im.NumDataAddr
          %assign im.AddrMap = im.AddrMap + data.BaseAddr
          %assign im.AddrMapComments = im.AddrMapComments + ...
            "%<im.NumDataAddr>: Block Parameter"
          %assign im.NumDataAddr = im.NumDataAddr + 1
          %%%%%%%%%%%%%%%%
          %% Parameter Name
          %assign paramName = mParam.Name
          %% Data Type
          %assign isPointer = TLC_FALSE
          %assign dataTypeMapIdx = ...
            FcnGetDataTypeMapIdx(dataTypeIdx,isComplex,isPointer,im)
          %% Dimensions
          %if data.NumDimensions > 2
            %assign DimensionsMap = []
            %foreach dimsIdx = SIZE(data.Dimensions,1)
              %assign DimensionsMap = DimensionsMap + data.Dimensions[dimsIdx]
            %endforeach
          %else
            %assign DimensionsMap = [%<data.NumRows>, %<data.NumCols>]
          %endif
          %assign dimMapIdx = FcnGetDimensionMapIdx(data, im, "", "")
          %% Fixed point
          %if isFixedPoint
            %assign fxpMapIdx = FcnGetFxpMapIdx(data,im)
          %else
            %assign fxpMapIdx = 0
          %endif
          %% Block Path
          %assign grBlkIdx = [%<sys>, %<blkIdx>]
          %assign blkPath = SLibMangledGrBlockPath(grBlkIdx)
          %assign blkPath = blkPath
          %assign blkPath = FEVAL ("repxpcxmlSchars",blkPath)
          %assign blkName = FEVAL ("repxpcxmlSchars",Block[blkIdx].SLName)
          %assign xpcblkName = blkPath
          %assign xpcblkName = FEVAL ("stripsysname",xpcblkName,%<lenName>)
          %% %assign xpcblkName = FEVAL ("repxpcxmlSchars",xpcblkName)
          %assign count = count + 1
          %if count == 1
            
            %<blkName>
            %<xpcblkName>
            %<blkPath>
            [%<isSrcBlk> %<niPorts> %<noPorts>]
          %endif
          
          %<paramName>
          %<DimensionsMap> %%<dim>[%<data.NumRows> %<data.NumCols>]
    %<slName>
          
         %%if paramIdx == NumParameters - 1
           %%
          %%endif
        %endforeach %%paramIdx = NumParameters
        %if count > 0
          
        %endif
      %endwith %%Block[blkIdx]
    %endforeach%%blkIdx = NumBlocks
  %endif %% InlineParameters == 0
  %%-----------------------------------------------------------
  %%sigNames-------------------------------
  %foreach blk = NumBlocks
    %if Block[blk].Virtual == 1
      %continue
    %endif
    %with Block[blk]
    %assign block = Block[blk]
    %assign isBusCreator = (Type == "BusCreator") ? 1 : 0
    %assign virtBlock = (block.Virtual > 0)
    %assign isSubsystem = (Type == "SubSystem") ? 1 : 0
    %assign isInport = (block.Type == "Inport") ? 1 : 0
    %assign isMdlRef = (Type == "ModelReference") ? 1 : 0
    %assign grIdx = [%<sys>,%<blk>]
    %%assign signalName = block.SLName
    %assign xpcsigName = SLibMangledGrBlockPath(grIdx)
    %assign xpcsigName = FEVAL ("repxpcxmlSchars",xpcsigName)
    %assign refblksysName = FEVAL ("repxpcxmlSchars",block.SLName)
    %if (isMdlRef)
      %assign mSysIdx = _blkref[0]
      %assign bIdx = _blkref[2]
      %if (bIdx>=0)
         %assign blkpath = FEVAL ("stripsysname",xpcsigName,%<lenName>)
         %assign mdlrefname = System[mSysIdx].Block[bIdx].ParamSettings.ReferencedModelName
         
             %<mdlrefname>
             %<blkpath>
             %<refblksysName>
    %<slName>
             
                   
             
         
         %%continue
      %endif %%if (bIdx>=0)
    %endif %%(isMdlRef)
    %if ((isSubsystem) || (isBusCreator)) %% || isMdlRef)
      %continue
    %endif
    %if Block[blk].Type == "Stateflow" && ...
      ISFIELD(Block[blk],"ChartData")
      %%---------------------------
      %assign sfblkName = FEVAL ("stripsysname",xpcsigName,%<lenName>)
      %assign sfcount = 0
      %foreach dwkIdx = Block[blk].ChartData.NumChartData
        %if Block[blk].ChartData.ChartData[dwkIdx].IsTestPoint
          %assign sfcount = sfcount + 1
          %if sfcount == 1
            
            %assign blksigName = block.SLName
            %<blksigName>
   %<xpcsigName>
          %endif
          %assign dwRec = CompiledModel.DWorks.DWork[Block[blk].DWork[dwkIdx]._idx]
          %assign chartData = Block[blk].ChartData.ChartData[dwkIdx]
          %assign structIdx = dwRec.MemoryMapIdx[0]
          %assign secIdx = dwRec.MemoryMapIdx[1]
          %assign dataIdx = dwRec.MemoryMapIdx[2]
          %if secIdx == -1
            %% Unstructured data
            %assign data = gmm.UnstructuredData.Data[dataIdx]
            %assign isComplex = data.IsComplex
            %if isComplex
              %continue
            %endif
            %assign dataTypeIdx = data.DataTypeIdx
            %assign dataAccess = data.Access
            %assign isPointer = dataAccess == "indirect" ? 1 : 0
          %elseif structIdx == -1
            %% Custom data
            %assign data = CompiledModel.GlobalMemoryMap.CustomData[secIdx].Data[dataIdx]
            %assign isComplex = data.IsComplex
            %if isComplex
              %continue
            %endif
            %assign dataTypeIdx = data.DataTypeIdx
            %assign dataAccess = data.Access
            %assign isPointer = dataAccess == "indirect" ? 1 : 0
          %else
            %% Structured data
            %assign section = gmm.StructuredData[structIdx].Section[secIdx]
            %%StructuredData[structIdx].Section[secIdx]
            %assign data = section.Data[dataIdx]
            %assign dataTypeIdx = section.DataTypeIdx
            %assign isComplex = section.IsComplex
            %if isComplex
              %continue
            %endif
            %assign isPointer = TLC_FALSE
            %assign dataAccess = "direct"
          %endif
          %% Block Path
          %assign grBlkIdx = [%<sys>, %<blk>]
          %assign blkPath = SLibMangledGrBlockPath(grBlkIdx) + ...
                "/" + chartData.Path
          %assign blkPath = FcnReturnCompliantBlockPath(blkPath, "Stateflow data")
          %% Signal Label
          %assign sigLabel = chartData.SFName
          %% Data Type
          %assign dataTypeMapIdx = ...
                FcnGetDataTypeMapIdx(dataTypeIdx,isComplex,isPointer,im)
          %% Dimension
          %assign dimMapIdx = FcnGetDimensionMapIdx(data,im,"","")
          %%------------------------------------------
          
          %assign sigWidth = data.NumRows * data.NumCols
          %assign sigpath = FEVAL ("repxpcxmlSchars","%<chartData.Path>")
          %assign sigName = FEVAL ("repxpcxmlSchars","%<chartData.SFName>")
          %assign resultN = FEVAL ("stripsysname",xpcsigName,%<lenName>)
          %assign resultN = FEVAL ("strrep",resultN,"'","''")
          %assign resultN = FEVAL ("repxpcxmlSchars",resultN)
          %assign sigpath = FEVAL ("repxpcxmlSchars","%<chartData.Path>")
          %assign sfxpcsigName = FEVAL("strcat","%<resultN>",":")
          %%assign sfxpcsigName = FEVAL("strcat","%<blksigName>",":")
          %assign sfxpcsigName = FEVAL("strcat","%<sfxpcsigName>","%<sigpath>")
          %<sigpath>
          %<sfxpcsigName>
          %<sigWidth>
          [%<data.NumRows>,%<data.NumCols>]
          %<sigName>
    true
    %<xpcsigName>
          %% %<resultN>
          %%
          
          %%------------------------------------------
        %endif %%Block[blkIdx].ChartData.ChartData[dwkIdx].IsTestPoint
      %endforeach
      %if sfcount > 0
        
      %endif
    %endif %%if Type == "Stateflow" && ISFIELD(block,"ChartData")
   %%---------------------------
    %foreach portIdx = NumDataOutputPorts
        %with DataOutputPort[portIdx]
          %if NumRegions > 1
            %continue
          %endif
         %if SignalLabel != ""
           %assign sigName = STRING(SignalLabel)
           %assign sigName = FEVAL ("repxpcxmlSchars",sigName)
         %else
           %assign sigName = ""
         %endif
          %foreach rgn = NumRegions
            %with Region[rgn]
              %assign mmi = MemoryMapIdx
              %if mmi[2] == -1
                %% Skip Local or reuse data
                %continue
              %else
                %assign structIdx = mmi[0]
                %assign secIdx = mmi[1]
                %assign dataIdx = mmi[2]
                %if secIdx == -1
                  %% Unstructured data
                  %assign data = CompiledModel.GlobalMemoryMap.UnstructuredData.Data[dataIdx]
                  %assign isComplex = data.IsComplex
                  %assign dataTypeIdx = data.DataTypeIdx
                  %assign dataAccess = data.Access
                  %assign isPointer = ...
                    dataAccess == "indirect" ? 1 : 0
                  %assign isStruct = data.IsStruct
                  %if isComplex || isStruct
                    %continue
                  %endif
                %elseif structIdx == -1
                  %% Custom data
                  %assign data = CompiledModel.GlobalMemoryMap.CustomData[secIdx].Data[dataIdx]
                  %assign isComplex = data.IsComplex
                  %assign dataTypeIdx = data.DataTypeIdx
                  %assign dataAccess = data.Access
                  %assign isPointer = ...
                    dataAccess == "indirect" ? 1 : 0
                  %assign constString = "(void *) "
                  %assign isStruct = data.IsStruct
                  %if isComplex || isStruct
                    %continue
                  %endif
                %else
                  %% Structured data
                  %assign section = ...
                    CompiledModel.GlobalMemoryMap.StructuredData[structIdx].Section[secIdx]
                  %assign data = section.Data[dataIdx]
                  %assign dataTypeIdx = section.DataTypeIdx
                  %assign isComplex = section.IsComplex
                  %assign isPointer = TLC_FALSE
                  %assign dataAccess = "direct"
                  %assign isStruct = data.IsStruct
                  %if isComplex || isStruct
                    %continue
                  %endif
                %endif
              %endif %% Non-Local Data
              %%if data.NumDimensions > 2
               %%continue
              %%endif
              %if data.NumDimensions > 2
                 %assign sigWidth = 1
                 %assign DimensionsMap = []
                 %foreach dimsIdx = SIZE(data.Dimensions,1)
                    %assign sigWidth = sigWidth * data.Dimensions[dimsIdx]
                    %assign DimensionsMap = DimensionsMap + data.Dimensions[dimsIdx]
                 %endforeach
              %else
                %assign sigWidth = data.NumRows * data.NumCols
                %assign DimensionsMap = [%<data.NumRows>, %<data.NumCols>]
              %endif
              %assign resultN = FEVAL ("stripsysname",xpcsigName,%<lenName>)
              %assign blksigName = FEVAL ("repxpcxmlSchars",block.SLName)
      
              %if (block.NumDataOutputPorts + block.NumFcnCallOutputPorts > 1)
                %assign resultN=resultN+"/p"+"%<portIdx+1>"
                %assign blksigName = blksigName + "/p" + "%<portIdx+1>"
              %endif
      
              %assign tp = "false"
              %if (TestPoint == 1)
                %assign tp = "true"
              %endif
              %assign resultN = FEVAL ("strrep",resultN,"'","''")
              %%assign resultN = FEVAL ("repxpcxmlSchars",resultN)
              
              %<blksigName>
              %<resultN>
              %<sigWidth>
              %<DimensionsMap>
     %<slName>
              %if sigName != ""
                %<sigName>
              %endif
     %<tp>
     %%<sigTestPoint>false
             %% %<resultN>
            %endwith %% Region[rgn]
            
          %endforeach %% rgn = NumRegions
        %endwith %% DataOutputPort[portIdx]
      %endforeach %% portIdx = NumDataOutputPorts
    %endwith %% Block[blk]
  %endforeach %% blk = NumBlocks
%endwith
%endfunction

  %else

      %<genxmlsubsys(bhm.Subsystem[0])>
      %<genxmlsubchilds(Subsystem[0],0)>
      %%---------model params
      %with GlobalMemoryMap
        %with ModelParameters
          %if NumModelParameters > 0

   Model Parameters
   Model Parameters

   [0 1 1]
          %endif
          %assign numpt=0
          %foreach paramIdx = NumModelParameters
            %assign param = ModelParameter[paramIdx]

              %assign structIdx = param.Instance[instIdx].MemoryMapIdx[0]
              %assign secIdx = param.Instance[instIdx].MemoryMapIdx[1]
              %assign dataIdx = param.Instance[instIdx].MemoryMapIdx[2]
              %if secIdx == -1 %% [-1 -1 dataIdx]
                %% Unstructured data
                %assign data = gmm.UnstructuredData.Data[dataIdx]
                %assign dTypeIdx = data.DataTypeIdx
                %assign isComplex = data.IsComplex
                %assign isStruct = data.IsStruct
                %if isComplex
                  %continue
                %endif
                %%if isStruct
                  %%continue
                %%endif
                %assign dataAccess = data.Access
                %assign section = data
              %elseif structIdx == -1 %% [-1 secIdx dataIdx]
                %% Custom data
                %assign data = CompiledModel.GlobalMemoryMap.CustomData[secIdx].Data[dataIdx]
                %assign dTypeIdx = data.DataTypeIdx
                %assign isComplex = data.IsComplex
                %assign isStruct = data.IsStruct
                %if isComplex %%|| isStruct
                  %continue
                %endif
                %assign dataAccess = data.Access
                %assign section = data
              %else
                %assign section = CompiledModel.GlobalMemoryMap.StructuredData[structIdx].Section[secIdx]
                %assign data = section.Data[dataIdx]
                %assign dTypeIdx = section.DataTypeIdx
                %assign isComplex = section.IsComplex
                %assign isStruct = data.IsStruct
                %%|| isStruct
                %if isComplex
                  %continue
                %endif
                %assign dataAccess = "direct"
              %endif
              %if data.Class == "other"
                %% Skip data of unknown layout
                %continue
              %endif
              %if dataAccess == "unknown"
                %% Skip data if DataAccess is unknown - R13 CSC's
                %continue
              %endif
              %assign dimsIdxCount = 0
              %assign DimensionsMap = []
              %assign DimensionsMapComments = []
              %%
              %% Skip certain kinds of data
              %%
              %if instIdx == 0
                %assign source = param.Instance[0].Source
                %%assign numVariableParams = numVariableParams + 1
              %elseif param.Instance[instIdx].Source != source
                %assign source = "multiple"
              %endif
              %if instIdx == 0
                %%assign class = FcnParamClass(data.Class)
                %%assign source = FcnParamSource(source)
                %if data.NumDimensions < 3
                  %assign dimsOffset = -1
                  %assign nCols = CAST("Number", data.NumCols)
                  %assign nRows = CAST("Number", data.NumRows)
                  %assign nDims = 2
                %else
                  %foreach dimsIdx = SIZE(data.Dimensions,1)
                    %assign DimensionsMap = DimensionsMap + data.Dimensions[dimsIdx]
                    %assign DimensionsMapComments = DimensionsMapComments + ...
                      "%<CAST("Number",dimsIdxCount+dimsIdx)>"
                  %endforeach
                  %assign dimsOffset = CAST("Number", dimsIdxCount)
                  %assign dimsIdxCount = dimsIdxCount + data.NumDimensions
                  %assign nCols = -1
                  %assign nRows = -1
                  %assign nDims = CAST("Number", data.NumDimensions)
                %endif
   
      %<param.Name>
                %if nDims < 3
          [%<nRows>,%<nCols>]
                %else
          %<DimensionsMap>
                %endif
   %<CompiledModel.Name>
   
              %endif%%instIdx == 0
            %endforeach%%instIdx = param.NumInstances
          %endforeach%%paramIdx = NumModelParameters
          %if NumModelParameters > 0

          %endif
        %endwith%%mdlparams
      %endwith%%globalMemMap
    %endwith %%bhm

  %endif %%isprotected
  %closefile Modelxml

%endwith %% CompiledModel
