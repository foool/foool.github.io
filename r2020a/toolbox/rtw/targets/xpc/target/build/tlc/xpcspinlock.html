%function LibGenTargetSpecificMutex(MutexOpType,semaphoreDW)
   %if !EXISTS("::xPCInlcudexPCImportshOnlyOnce")
      %<LibAddToCommonIncludes("xpcimports.h")>
      %assign ::xPCInlcudexPCImportshOnlyOnce = 1
  %endif
  %switch (MutexOpType)
    %case 0
      %return "%<semaphoreDW> = TBCreateSpin(); /n"
    %case 1
      %return "TBTakeSpinFast(%<semaphoreDW>); /n"
    %case 2
      %return "TBGiveSpinFast(%<semaphoreDW>); /n"
    %case 3
      %return "TBDeleteSpin(%<semaphoreDW>); /n"
    %default
      %return " error out /n"
  %endswitch
%endfunction
 
 
%function LibGenTargetSpecificSynchro(SynchroOpType,synchroDW, initVal) void
   %if !EXISTS("::xPCInlcudexPCImportshOnlyOnce")
      %<LibAddToCommonIncludes("xpcimports.h")>
      %assign ::xPCInlcudexPCImportshOnlyOnce = 1
  %endif
  %switch (SynchroOpType)
    %case 0
      %assign semaName = "xPCSema"
      %return "%<synchroDW> = xpceCreateSemaphore(2, %<initVal>, semaName ); /n"
    %case 1
      %return "xpceWaitSemaphore(%<synchroDW>); /n"
    %case 2
      %return "xpceSignalSemaphore(%<synchroDW>); /n"
    %case 3
      %return "xpceDeleteSemaphore(%<synchroDW>); /n"
    %default
      %return " error out /n"
  %endswitch
%endfunction
