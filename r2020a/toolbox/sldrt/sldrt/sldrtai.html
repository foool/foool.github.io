%implements "sldrtai" "C"
 
%%includeSimulinkDesktopReal-Timelibrary
%include "sldrtlib.tlc"
 
 
 
%%Function:BlockInstanceSetup====================================================
%%Abstract:
%%Registerboardtoboardtable
%%
%function BlockInstanceSetup(block, system) void
 
%<SLDRTRegisterBoard(block)>
 
%endfunction %% BlockInstanceSetup
 
 
 
%%Function:Outputs===========================================================
%%Abstract:
%%Readreal-timeanaloginputsbycalling:
%%RTBIO_DriverIO(adapter,ANALOGINPUT,IOREAD,n,ch,val,parm);
%%
%function Outputs(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  {
    %assign n = LibBlockOutputSignalWidth(0)
    %assign chanptr = LibBlockParameterAddr(Channels, "", "", 0)
    %assign rangemode = LibBlockParameter(RangeMode, "", "", 0)
    %assign voltidx = LibBlockParameter(VoltRange, "", "", 0)
    %assign valptr = SLDRTAllocOutputBuffer(0, n)
%%
    ANALOGIOPARM parm;
    parm.mode = (RANGEMODE) %<rangemode>;
    parm.rangeidx = %<voltidx>;
    RTBIO_DriverIO(%<block.SLDRTBoardIdx>, ANALOGINPUT, IOREAD, %<n>, %<chanptr>, %<valptr>, &parm);
    %<SLDRTCopyOutputBuffer(block, 0, n)>
 
    %% handle Missed Ticks port
    %assign dummy = SLDRTNumOutputPortsMissedTicks(block)
  }
%%
%endfunction %% Outputs
