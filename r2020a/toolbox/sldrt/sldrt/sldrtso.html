%%Function:BlockInstanceSetup====================================================
%%Abstract:
%%Registerboardtoboardtable
%%
%function BlockInstanceSetup(block, system) void
 
%<SLDRTRegisterBoard(block)>
 
%endfunction %% BlockInstanceSetup
 
 
 
%%Function:Start==============================================================
%%Abstract:
%%Writeinitialvalue
%%
%function Start(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  %if ISFIELD(block, "InitialValue")
    RTBIO_DriverIO(%<block.SLDRTBoardIdx>, STREAMOUTPUT, IOWRITE, sizeof("%<InitialValue>")-1, NULL, (double*) "%<InitialValue>", NULL);
  %else
    /* no initial value should be set */
  %endif
%%
%endfunction %% Start
 
 
 
%%Function:Outputs===========================================================
%%Abstract:
%%Writeoutputsoneachtimestep
%%
%function Outputs(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  %% handle Missed Ticks port
  %assign numports = SLDRTNumOutputPortsMissedTicks(block)
 
  %% don't do anything if no output ports
  %if numports <= 0
    /* no code required */
    %return
  %endif
 
  {
    %% update Data Ready and Data Error outputs
    int status = RTBIO_DriverIO(%<block.SLDRTBoardIdx>, STREAMOUTPUT, IOSTATUS, %<MessageSize>, NULL, NULL, NULL);
    %if ShowError
      %assign numports = numports-1
      %<LibBlockOutputSignal(numports, "", "", 0)> = (status>>1) & 0x1; /* Data Error port */
    %endif
    %if numports > 0
      %<LibBlockOutputSignal(numports-1, "", "", 0)> = status & 0x1; /* Data Ready port */
    %endif
  }
%%
%endfunction %% Outputs
 
 
 
%%Function:Update===========================================================
%%Abstract:
%%Updatestatesoneachtimestep
%%
%function Update(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  {
    %assign val = ""
    %assign dtidx = 0
    %foreach portIdx = LibBlockNumInputPorts(block)
      %if FmtDataTypes[dtidx] == "string"
        %assign val = val + ", (const char_T*) " + LibBlockInputSignalAddr(portIdx, "", "", 0)
        %assign dtidx = dtidx + 1
      %else
        %foreach sigIdx = LibBlockInputSignalWidth(portIdx)
          %assign val = val + ", (" + FmtDataTypes[dtidx] + ") " + LibBlockInputSignal(portIdx, "", "", sigIdx)
          %assign dtidx = dtidx + 1
        %endforeach
      %endif
    %endforeach
 
    char_T outstring[%<MessageSize>];
    int n = snprintf(outstring, %<MessageSize>, "%<FormatString>"%<val>);
    RTBIO_DriverIO(%<block.SLDRTBoardIdx>, STREAMOUTPUT, IOWRITE, n, NULL, (double*) outstring, NULL);
  }
%%
%endfunction %% Update
 
 
 
%%Function:Terminate==============================================================
%%Abstract:
%%Writefinalvalue
%%
%function Terminate(block, system) Output
%%
  %if ISFIELD(block, "FinalValue")
    RTBIO_DriverIO(%<block.SLDRTBoardIdx>, STREAMOUTPUT, IOWRITE, sizeof("%<FinalValue>")-1, NULL, (double*) "%<FinalValue>", NULL);
  %else
    /* no final value should be set */
  %endif
%%
%endfunction %% Terminate
