%%Function:RTWinPacketOutputEndianCopy===========================================
%%Abstract:
%%Copyavaluetooutputpacket,takingendianintoaccount
%%
%function RTWinPacketOutputEndianCopy(block, srcdtid, src, dtid, byteSize) Output
%%
  {
    %% Standard numeric data type IDs:
    %% 0: double, 1: single, 2: int8, 3: uint8, 4: int16, 5: uint16, 6: int32, 7: uint32
    %assign dt = LibGetDataTypeNameFromId(dtid)
    %assign srcdt = LibGetDataTypeNameFromId(srcdtid)
 
    %% minimum and maximum value of target data type
    %assign dtMaxValue = "%dt>")>"
    %if dtid==3 || dtid==5 || dtid==7 %% minimum for unsigned integer types is unsigned zero
      %assign dtMinValue = "0U"
    %else
      %assign dtMinValue = "%dt>")>"
    %endif
 
    %% no type conversion if data types match
    %if srcdtid==dtid
      %<dt> pktout = %<src>;
 
    %% handle conversions from double and single to integers
    %% single to int32 and uint32 must be converted via real_T to exactly compare to 32-bit limits
    %elseif (srcdtid==0 || srcdtid==1) && dtid>=2 && dtid<=7
      %if srcdtid==1 && (dtid==6 || dtid==7)
        real_T pktin = %<src>;
      %else
        %<srcdt> pktin = %<src>;
      %endif
      %<dt> pktout = pktin>=%<dtMaxValue> ? %<dtMaxValue> : (pktin<=%<dtMinValue> ? %<dtMinValue> : (%<dt>) pktin);
 
    %% handle signed integer target data types
    %elseif (dtid==2 || dtid==4 || dtid==6)
 
      %if (srcdtid==2 || srcdtid==4 || srcdtid==6) %% conversion from signed data type
        %if srcdtid>dtid
          %<srcdt> pktin = %<src>;
          %<dt> pktout = pktin>=%<dtMaxValue> ? %<dtMaxValue> : (pktin<=%<dtMinValue> ? %<dtMinValue> : (%<dt>) pktin);
        %else
          %<dt> pktout = (%<dt>) %<src>;
        %endif
 
      %else %% conversion from unsigned data type
        %if srcdtid>dtid
          %<srcdt> pktin = %<src>;
          %<dt> pktout = pktin>=%<dtMaxValue> ? %<dtMaxValue> : (%<dt>) pktin;
        %else
          %<dt> pktout = (%<dt>) %<src>;
        %endif
      %endif
 
    %% handle unsigned integer target data types
    %elseif (dtid==3 || dtid==5 || dtid==7)
 
      %if (srcdtid==2 || srcdtid==4 || srcdtid==6) %% conversion from signed data type
        %<srcdt> pktin = %<src>;
        %if srcdtid>dtid
          %<dt> pktout = pktin>=%<dtMaxValue> ? %<dtMaxValue> : (pktin<=0 ? 0 : (%<dt>) pktin);
        %else
          %<dt> pktout = pktin<0 ? 0 : (%<dt>) pktin;
        %endif
 
      %else %% conversion from unsigned data type
        %if srcdtid>dtid
          %<srcdt> pktin = %<src>;
          %<dt> pktout = pktin>=%<dtMaxValue> ? %<dtMaxValue> : (%<dt>) pktin;
        %else
          %<dt> pktout = (%<dt>) %<src>;
        %endif
      %endif
 
    %% plain typecast for other data type combinations
    %else
      %<dt> pktout = (%<dt>) %<src>;
    %endif
       
    %% copy the value if little endian or if single byte
    %if ByteOrder==1 || byteSize==1
      *outdp.p_%<dt>++ = pktout;
 
    %% reverse the byte order of the value if big endian
    %else
      /* %<dt>, big endian */
      uint8_T* pktoutptr = ((uint8_T*) &pktout) + %<byteSize>;
      %if byteSize>RollThreshold
        int i;
        for (i=0; i<%<byteSize>; ++i)
          *outdp.p_uint8_T++ = *--pktoutptr;
      %else
        %foreach byteIdx = byteSize
          *outdp.p_uint8_T++ = *--pktoutptr;
        %endforeach
      %endif
 
    %endif
  }
%%
%endfunction %% RTWinPacketOutputEndianCopy
 
 
 
%%Function:RTWinPacketOutputSendParameterValue===================================
%%Abstract:
%%Sendparametervalueaspacketoutput
%%
%function RTWinPacketOutputSendParameterValue(block, value) Output
%%
  /* %<Type> Block: %<Name> */
  %assign valsize = LibBlockParameterWidth(value)
  %if valsize==0
    /* no initial value should be set */
    %return
  %endif
 
  {
    const double* parm = %<LibBlockParameterBaseAddr(value)>;
    char outdata[%<PacketSize>];
    RTWin_ANYTYPEPTR outdp;
    outdp.p_uint8_T = outdata;
 
    %% loop over ports
    %foreach portIdx = LibBlockNumInputPorts(block)
      %assign dtid = PktDataTypeIds[portIdx]
      %assign byteSize = LibGetDataTypeSLSizeFromId(PktDataTypeIds[portIdx])
 
      %% roll over parameter elements
      %roll sigIdx = DataInputPort[portIdx].RollRegions, lcv = RollThreshold, block, "Roller", ""
 
        %% copy the value, taking endian into account
        %<RTWinPacketOutputEndianCopy(block, 0, "*parm++", dtid, byteSize)>
 
      %endroll
    %endforeach
 
    %% actually write the packet
    %assign pktidptr = LibBlockParameterAddr(PacketID, "", "", 0)
    RTBIO_DriverIO(%<block.SLDRTBoardIdx>, STREAMOUTPUT, IOWRITE, %<PacketSize>, %<pktidptr>, (double*) outdata, NULL);
  }
%%
%endfunction %% RTWinPacketOutputSendParameterValue

%endfunction %% Start
 
 
 
%%Function:Outputs===========================================================
%%Abstract:
%%Writeoutputsoneachtimestep
%%
%function Outputs(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  %% handle Missed Ticks port
  %assign numports = SLDRTNumOutputPortsMissedTicks(block)
 
  %% don't do anything if no output ports
  %if numports <= 0
    /* no code required */
    %return
  %endif
 
  %if PacketSize>0
    %assign pktwritesize = PacketSize
  %else
    %assign pktwritesize = 1
  %endif
 
  {
    %% update Data Ready and Data Error outputs
    int status = RTBIO_DriverIO(%<block.SLDRTBoardIdx>, STREAMOUTPUT, IOSTATUS, %<pktwritesize>, NULL, NULL, NULL);
    %if ShowError
      %assign numports = numports-1
      %<LibBlockOutputSignal(numports, "", "", 0)> = (status>>1) & 0x1; /* Data Error port */
    %endif
    %if numports > 0
      %<LibBlockOutputSignal(numports-1, "", "", 0)> = status & 0x1; /* Data Ready port */
    %endif
  }
%%
%endfunction %% Outputs
 
 
 
%%Function:Update===========================================================
%%Abstract:
%%Updatestatesoneachtimestep
%%
%function Update(block, system) Output
%%
  %% don't do anything if no data to send
  %if PacketSize <= 0
    %return
  %endif
 
  /* %<Type> Block: %<Name> */
  {
    %if !LibIsFundamentalBuiltInDataType(PktDataTypeIds[0]) %% CAN_MESSAGE
 
      const CAN_MESSAGE* msg = (const CAN_MESSAGE*) %<LibBlockInputSignalAddr(0, "", "", 0)>;
 
      %% read packet ID from CAN_MESSAGE
      int32_T pktid = msg->Extended ? -msg->ID : (msg->ID | 0x800);
      %assign pktidptr = "&pktid"
 
      %assign outdatalen = "msg->Length"
      %assign outdataptr = "(double*) (&msg->Data)"
 
    %else
      %% read packet ID from parameter
      %assign pktidptr = LibBlockParameterAddr(PacketID, "", "", 0)
 
      %% declare output data buffer
      %assign outdatalen = PacketSize
      %assign outdataptr = "(double*) outdata"
      uint8_T outdata[%<PacketSize>];
 
      RTWin_ANYTYPEPTR outdp;
      outdp.p_uint8_T = outdata;
 
      %% loop over ports
      %foreach portIdx = LibBlockNumInputPorts(block)
        %assign dtid = PktDataTypeIds[portIdx]
        %assign byteSize = LibGetDataTypeSLSizeFromId(PktDataTypeIds[portIdx])
 
        %% roll over signal elements
        %assign rollVar = SPRINTF("u%d", portIdx)
        %roll sigIdx = DataInputPort[portIdx].RollRegions, lcv = RollThreshold, block, "Roller", [ %<rollVar> ]
 
          %% copy the value, taking endian into account
          %<RTWinPacketOutputEndianCopy(block, LibBlockInputSignalDataTypeId">LibBlockInputSignalDataTypeId(portIdx), LibBlockInputSignal(portIdx, "", lcv, sigIdx), dtid, byteSize)>
 
        %endroll
      %endforeach
 
    %endif
 
 
    %% actually write the packet
    RTBIO_DriverIO(%<block.SLDRTBoardIdx>, STREAMOUTPUT, IOWRITE, %<outdatalen>, %<pktidptr>, %<outdataptr>, NULL);
  }
%%
%endfunction %% Update
