%%Function:Outputs===========================================================
%%Abstract:
%%Writeoutputsoneachtimestep
%%
%function Outputs(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  {
    %% read data and status from driver
    char indata[%];
    %if ISFIELD(block, "Terminator")
      %assign terms = ""
      %foreach idx = SIZE(Terminator, 1)
        %assign terms = terms + Terminator[idx] + "//0"
      %endforeach
      int status;
      const char* terminators = "%<terms>";
      memset(indata, 0, sizeof(indata)); %% zero message buffer to ensure the variable-length message is zero-terminated
      status = RTBIO_DriverIO(%<block.SLDRTBoardIdx>, STREAMINPUT, IOREADWITHRESET, %<MessageSize>, NULL, (double*) indata, terminators);
    %else
      int status = RTBIO_DriverIO(%<block.SLDRTBoardIdx>, STREAMINPUT, IOREAD, %<MessageSize>, NULL, (double*) indata, NULL);
      indata[%<MessageSize>] = '/0'; %% zero-terminate the fixed-length message
    %endif
 
    %% handle Missed Ticks port
    %assign numports = SLDRTNumOutputPortsMissedTicks(block)
 
    %% update Data Ready and Data Error outputs
    %if ShowError
      %assign numports = numports-1
      %<LibBlockOutputSignal(numports, "", "", 0)> = (status>>1) & 0x1; /* Data Error port */
    %endif
    %if ShowReady
      %assign numports = numports-1
      %<LibBlockOutputSignal(numports, "", "", 0)> = status & 0x1; /* Data Ready port */
    %endif
 
    %% update outputs only if data ready
    if (status & 0x1)
    {
      %% declare and initialize variables
      %assign dtidx = 0
      %assign vars = ""
      %foreach portIdx = numports
        %if FmtDataTypes[dtidx] == "string"
      %assign vars = vars + ", " + "%<LibBlockOutputSignalAddr(portIdx, "", "", 0)>"
          %assign dtidx = dtidx + 1
        %else
      %foreach sigIdx = LibBlockOutputSignalWidth(portIdx)
         %<SPRINTF("%s var%d;", FmtDataTypes[dtidx], dtidx)>
         %assign vars = vars + SPRINTF(", &var%d", dtidx)
         %assign dtidx = dtidx + 1
      %endforeach
        %endif
      %endforeach
 
      %% convert message to numeric variables
      int nscan = sscanf(indata, "%<FormatString>"%<vars>);
 
      %% copy the converted values to output ports
      %assign dtidx = 0
      %foreach portIdx = numports
        %if FmtDataTypes[dtidx] == "string"
      %assign dtidx = dtidx+1 %% string already assigned
        %else
          %assign dt = LibBlockOutputSignalDataTypeName(portIdx, tRealPart)
      %foreach sigIdx = LibBlockOutputSignalWidth(portIdx)
        if (nscan > %<dtidx>)
          %<LibBlockOutputSignal(portIdx, "", "", sigIdx)> = (%<dt>) var%<dtidx>;
        %assign dtidx = dtidx+1
      %endforeach
        %endif
      %endforeach
    }
  }
%%
%endfunction %% Outputs
