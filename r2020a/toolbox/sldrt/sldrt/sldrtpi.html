%%Function:Outputs===========================================================
%%Abstract:
%%Writeoutputsoneachtimestep
%%
%function Outputs(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  {
    %% read data and status from driver
    %if PacketSize>0
      %assign pktreadsize = PacketSize
      %assign indataptr = "(double*) indata"
      uint8_T indata[%<PacketSize>];
    %else
      %assign pktreadsize = 1
      %assign indataptr = "NULL"
    %endif
    %assign pktidptr = LibBlockParameterAddr(PacketID, "", "", 0)
 
    %if ShowTimestamp || IsCAN_MESSAGE
      CANMSGPARM canmp;
      %assign canmpptr = "&canmp"
    %else
      %assign canmpptr = "NULL"
    %endif
    int status = RTBIO_DriverIO(%<block.SLDRTBoardIdx>, STREAMINPUT, IOREAD, %<pktreadsize>, %<pktidptr>, %<indataptr>, %<canmpptr>);
 
    %% handle Missed Ticks port
    %assign numports = SLDRTNumOutputPortsMissedTicks(block)
 
    %% Timestamp port follows Data Ready and Data Error
    %if ShowTimestamp
      %assign numports = numports-1
      %assign timestampport = numports
    %endif
 
    %% update Data Ready and Data Error outputs
    %if ShowError
      %assign numports = numports-1
      %<LibBlockOutputSignal(numports, "", "", 0)> = (status>>1) & 0x1; /* Data Error port */
    %endif
    %if ShowReady
      %assign numports = numports-1
      %<LibBlockOutputSignal(numports, "", "", 0)> = status & 0x1; /* Data Ready port */
    %endif
 
    %% update outputs only if data ready
    %if PacketSize>0
      if (status & 0x1)
      {
 
      %if IsCAN_MESSAGE
 
        /* fill in the CAN_MESSAGE structure */
        CAN_MESSAGE* msg = (CAN_MESSAGE*) %<LibBlockOutputSignalAddr(0, "", "", 0)>;
        int32_T pktid = %<LibBlockParameter(PacketID, "", "", 0)>;
        msg->Extended = pktid<=0; /* extended ID coded as negative */
        msg->Length = canmp.msglen;
        msg->Remote = 0;
        msg->Error = (status>>1) & 0x1;
        msg->ID = pktid>0 ? (pktid & 0x7FF) : ((-pktid) & 0x1FFFFFFF); /* extended ID coded as negative */
        msg->Timestamp = canmp.timestamp;
        memcpy(&(msg->Data), indata, msg->Length);
 
      %else
 
        RTWin_ANYTYPEPTR indp;
        indp.p_uint8_T = indata;
 
        %% loop over ports
        %foreach portIdx = numports
          %assign dt = LibBlockOutputSignalDataTypeName(portIdx, tRealPart)
          %assign byteSize = LibGetDataTypeSLSizeFromId(LibBlockOutputSignalDataTypeId(portIdx))
 
          %% roll over signal elements
          %assign rollVar = SPRINTF("y%d", portIdx)
          %roll sigIdx = DataOutputPort[portIdx].SignalOffset, lcv = RollThreshold, block, "Roller", [ %<rollVar> ]
 
            %% copy the values if little endian or if single byte
            %if ByteOrder==1 || byteSize==1
              %<LibBlockOutputSignal(portIdx, "", lcv, sigIdx)> = *indp.p_%<dt>++;
 
            %% reverse the byte order of the value if big endian
            %else
            {
              /* %<dt>, big endian */
              uint8_T* blkoutptr = ((uint8_T *) %<LibBlockOutputSignalAddr(portIdx, "", lcv, sigIdx)>) + %<byteSize>;
              %if byteSize>RollThreshold
                int i;
                for (i=0; i<%<byteSize>; ++i)
                  *--blkoutptr = *indp.p_uint8_T++;
              %else
                %foreach byteIdx = byteSize
                  *--blkoutptr = *indp.p_uint8_T++;
                %endforeach
              %endif
            }
 
            %endif
          %endroll
        %endforeach
 
      %endif
 
        %if ShowTimestamp
          %<LibBlockOutputSignal(timestampport, "", "", 0)> = canmp.timestamp; %% Timestamp port
        %endif
      }
    %endif
  }
%%
%endfunction %% Outputs
