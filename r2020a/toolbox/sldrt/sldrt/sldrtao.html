%implements "sldrtao" "C"
 
%%includeSimulinkDesktopReal-Timelibrary
%include "sldrtlib.tlc"
 
 
 
%%Function:BlockInstanceSetup====================================================
%%Abstract:
%%Registerboardtoboardtable
%%
%function BlockInstanceSetup(block, system) void
 
%<SLDRTRegisterBoard(block)>
 
%endfunction %% BlockInstanceSetup
 
 
 
%%Function:RTWinAnalogOutput====================================================
%%Abstract:
%%Writereal-timeanalogoutputsbycalling:
%%RTBIO_DriverIO(adapter,ANALOGOUTPUT,IOWRITE,n,ch,val,parm);
%%
%function RTWinAnalogOutput(block, n, val) Output
%%
  %assign chanptr = LibBlockParameterAddr(Channels, "", "", 0)
  %assign rangemode = LibBlockParameter(RangeMode, "", "", 0)
  %assign voltidx = LibBlockParameter(VoltRange, "", "", 0)
%%
  {
    ANALOGIOPARM parm;
 
    parm.mode = (RANGEMODE) %<rangemode>;
    parm.rangeidx = %<voltidx>;
    RTBIO_DriverIO(%<block.SLDRTBoardIdx>, ANALOGOUTPUT, IOWRITE, %<n>, %<chanptr>, %<val>, &parm);
  }
%%
%endfunction %% RTWinAnalogOutput
 
 
 
%%Function:Start==============================================================
%%Abstract:
%%Writeinitialvalue
%%
%function Start(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  %assign valsize = LibBlockParameterWidth(InitialValue)
  %if valsize==0
 
  /* no initial value should be set */
 
  %else
  {
    %assign n = LibBlockParameterWidth(Channels)
    %assign val = SLDRTExpandParameter(block, InitialValue, n, "aoval")
    %<RTWinAnalogOutput(block, n, val)>
  }
  %endif
%%
%endfunction %% Start
 
 
 
%%Function:Outputs===========================================================
%%Abstract:
%%Writeoutputsoneachtimestep
%%
%function Outputs(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  {
    %assign n = LibBlockParameterWidth(Channels)
    %assign val = SLDRTExpandInput(block, 0, n, "aoval")
    %<RTWinAnalogOutput(block, n, val)>
 
    %% handle Missed Ticks port
    %assign dummy = SLDRTNumOutputPortsMissedTicks(block)
  }
%%
%endfunction %% Outputs
 
 
 
%%Function:Terminate==============================================================
%%Abstract:
%%Writefinalvalue
%%
%function Terminate(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  %assign valsize = LibBlockParameterWidth(FinalValue)
  %if valsize==0
 
  /* no final value should be set */
 
  %else
  {
    %assign n = LibBlockParameterWidth(Channels)
    %assign val = SLDRTExpandParameter(block, FinalValue, n, "aoval")
    %<RTWinAnalogOutput(block, n, val)>
  }
  %endif
%%
%endfunction %% Terminate
