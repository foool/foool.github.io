%implements "sldrtqo" "C"

%include "sldrtlib.tlc"
 
 
 
%%Function:BlockInstanceSetup====================================================
%%Abstract:
%%Registerboardtoboardtable
%%
%function BlockInstanceSetup(block, system) void
 
%<SLDRTRegisterBoard(block)>
 
%endfunction %% BlockInstanceSetup
 
 
 
%%Function:RTWinFrequencyOutput====================================================
%%Abstract:
%%Writereal-timefrequencyoutputsbycalling:
%%RTBIO_DriverIO(adapter,FREQUENCYOUTPUT,IOWRITE,n,ch,fval,dval);
%%
%function RTWinFrequencyOutput(block, n, fval, dval) Output
%%
  %assign chanptr = LibBlockParameterAddr(Channels, "", "", 0)
  RTBIO_DriverIO(%<block.SLDRTBoardIdx>, FREQUENCYOUTPUT, IOWRITE, %<n>, %<chanptr>, %<fval>, %<dval>);
%%
%endfunction %% RTWinFrequencyOutput
 
 
 
%%Function:Start==============================================================
%%Abstract:
%%Writeinitialvalue
%%
%function Start(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  %if (FrequencySource==0 || LibBlockParameterWidth(Frequency)==0) ...
      && (DutySource==0 || LibBlockParameterWidth(Duty)==0)
 
  /* no initial value should be set */
 
  %else
  {
    %assign n = LibBlockParameterWidth(Channels)
    %assign fval = SLDRTExpandParameter(block, Frequency, n, "qofval")
    {
      %assign dval = SLDRTExpandParameter(block, Duty, n, "qodval")
      %<RTWinFrequencyOutput(block, n, fval, dval)>
    }
  }
  %endif
%%
%endfunction %% Start
 
 
 
%%Function:Outputs===========================================================
%%Abstract:
%%Writeoutputsoneachtimestep
%%
%function Outputs(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  {
    %assign n = LibBlockParameterWidth(Channels)
    %if FrequencySource==0
      %assign fval = SLDRTExpandParameter(block, Frequency, n, "qofval")
      %assign dutyidx = 0
    %else
      %assign fval = SLDRTExpandInput(block, 0, n, "qofval")
      %assign dutyidx = 1
    %endif
    {
      %if DutySource==0
        %assign dval = SLDRTExpandParameter(block, Duty, n, "qodval")
      %else
    %assign dval = SLDRTExpandInput(block, dutyidx, n, "qodval")
      %endif
      %<RTWinFrequencyOutput(block, n, fval, dval)>
    }
 
    %% handle Missed Ticks port
    %assign dummy = SLDRTNumOutputPortsMissedTicks(block)
  }
%%
%endfunction %% Outputs
 
 
 
%%Function:Terminate==============================================================
%%Abstract:
%%Writefinalvalue
%%
%function Terminate(block, system) Output
%%
  /* %<Type> Block: %<Name> */
  %if (FrequencySource==0 || LibBlockParameterWidth(FrequencyFinalValue)==0) ...
      && (DutySource==0 || LibBlockParameterWidth(DutyFinalValue)==0)
 
  /* no final value should be set */
 
  %else
  {
    %assign n = LibBlockParameterWidth(Channels)
    %if FrequencySource==0
      %assign fval = SLDRTExpandParameter(block, Frequency, n, "qofval")
    %else
      %assign fval = SLDRTExpandParameter(block, FrequencyFinalValue, n, "qofval")
    %endif
    {
      %if DutySource==0
        %assign dval = SLDRTExpandParameter(block, Duty, n, "qodval")
      %else
        %assign dval = SLDRTExpandParameter(block, DutyFinalValue, n, "qodval")
      %endif
      %<RTWinFrequencyOutput(block, n, fval, dval)>
    }
  }
  %endif
%%
%endfunction %% Terminate
